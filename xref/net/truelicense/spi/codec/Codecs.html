6   package net.truelicense.spi.codec;
7   
8   import net.truelicense.api.codec.Codec;
9   import net.truelicense.obfuscate.Obfuscate;
10  
11  import java.nio.charset.Charset;
12  import java.nio.charset.StandardCharsets;
13  import java.util.Optional;
14  import java.util.regex.Matcher;
15  import java.util.regex.Pattern;
16  
17  /**
18   * Provides common {@link Codec} functions.
19   *
20   * @author Christian Schlichtherle
21   */
22  public class Codecs {
23  
24      @Obfuscate private static final String SEVEN_BIT = "7bit";
25      @Obfuscate private static final String EIGHT_BIT = "8bit";
26      @Obfuscate private static final String QUOTED_PRINTABLE = "quoted-printable";
27      @Obfuscate private static final String BASE64 = "base64";
28  
29      // Not perfect, but should work for all valid input.
30      @Obfuscate private static final String CHARSET_REGEXP =
31              "\\s*charset\\s*=\\s*(?:\"([^\"]*)\"|([^\\s()<>@,;:\\\\\"/\\[\\]?=]+))";
32  
33      private static final Pattern CHARSET_PATTERN = Pattern.compile(CHARSET_REGEXP, Pattern.CASE_INSENSITIVE);
34  
35      /**
36       * Figures the content transfer charset which is used by the given codec.
37       * <ol>
38       * <li>
39       * First, the codec's
40       * {@linkplain Codec#contentTransferEncoding Content-Transfer-Encoding}
41       * gets tested:
42       * If it is {@code "7bit"}, {@code "quoted-printable"} or {@code "base64"}
43       * (ignoring case), then the returned charset is US-ASCII.
44       * <li>
45       * Otherwise, if the codec's Content-Transfer-Encoding is {@code "8bit"}
46       * (ignoring case), then the codec's {@link Codec#contentType Content-Type}
47       * gets tested:
48       * <ol>
49       * <li>
50       * If the codec's Content-Type specifies a {@code charset} parameter
51       * (ignoring case), then an attempt is made to load this charset.
52       * Upon success, this charset gets returned.
53       * <li>
54       * If the codec's Content-Type does not specify a {@code charset} parameter,
55       * then UTF-8 gets returned.
56       * This ensures compatibility with JSON.
57       * </ol>
58       * <li>
59       * Finally, in all other cases, the optional return value is empty in order to specify that the codec doesn't
60       * produce text or specifies an invalid charset name or an unknown charset.
61       * </ol>
62       *
63       * @param  codec the codec to test.
64       * @return The optional content transfer charset which is used by the given codec.
65       *         Maybe empty if the codec doesn't produce text or specifies an invalid charset name or an unknown charset.
66       * @since TrueLicense 3.1.0
67       */
68      public static Optional<Charset> charset(final Codec codec) {
69          final String encoding = codec.contentTransferEncoding();
70          if (SEVEN_BIT.equalsIgnoreCase(encoding) ||
71                  QUOTED_PRINTABLE.equalsIgnoreCase(encoding) ||
72                  BASE64.equalsIgnoreCase(encoding)) {
73              return Optional.of(StandardCharsets.US_ASCII);
74          } else if (EIGHT_BIT.equalsIgnoreCase(encoding)) {
75              final Matcher matcher = CHARSET_PATTERN.matcher(codec.contentType());
76              if (matcher.find()) {
77                  assert 2 == matcher.groupCount();
78                  try {
79                      return Optional.of(Optional
80                              .ofNullable(matcher.group(1))
81                              .map(Charset::forName)
82                              .orElseGet(() -> Charset.forName(matcher.group(2))));
83                  } catch (IllegalArgumentException ignored) {
84                      return Optional.empty();
85                  }
86              } else {
87                  return Optional.of(StandardCharsets.UTF_8);
88              }
89          }
90          return Optional.empty();
91      }
92  
93      private Codecs() { }
94  }
