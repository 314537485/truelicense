6   package net.truelicense.spi.io;
7   
8   import net.truelicense.api.io.Store;
9   
10  import java.io.*;
11  import java.util.Optional;
12  import java.util.prefs.BackingStoreException;
13  import java.util.prefs.Preferences;
14  
15  import static java.util.Objects.requireNonNull;
16  
17  /**
18   * A preferences (node) store.
19   *
20   * @author Christian Schlichtherle
21   */
22  final class PreferencesStore implements Store {
23  
24      private final Preferences prefs;
25      private final String key;
26  
27      PreferencesStore(final Preferences prefs, final String key) {
28          this.prefs = requireNonNull(prefs);
29          this.key = requireNonNull(key);
30      }
31  
32      @Override
33      public InputStream input() throws IOException {
34          return new ByteArrayInputStream(data());
35      }
36  
37      @Override
38      public OutputStream output() throws IOException {
39          return new ByteArrayOutputStream(BUFSIZE) {
40              @Override
41              public void close() throws IOException {
42                  data(toByteArray());
43              }
44          };
45      }
46  
47      @Override
48      public void delete() throws IOException {
49          data();
50          prefs.remove(key);
51          sync();
52      }
53  
54      @Override
55      public boolean exists() { return optData().isPresent(); }
56  
57      @SuppressWarnings("LoopStatementThatDoesntLoop")
58      private byte[] data() throws IOException {
59          return optData()
60                  .orElseThrow(() -> new FileNotFoundException(
61                          "Cannot locate the key \"" + key + "\" in the " +
62                                  (prefs.isUserNode() ? "user" : "system") +
63                                  " preferences node for the absolute path \"" +
64                                  prefs.absolutePath() + "\"."));
65      }
66  
67      private void data(byte[] data) throws IOException {
68          prefs.putByteArray(key, data);
69          sync();
70      }
71  
72      private Optional<byte[]> optData() {
73          return Optional.ofNullable(prefs.getByteArray(key, null));
74      }
75  
76      private void sync() throws IOException {
77          try { prefs.flush(); }
78          catch (final BackingStoreException e) { throw new IOException(e); }
79      }
80  }
