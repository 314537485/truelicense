6   package net.truelicense.spi.io;
7   
8   import net.truelicense.api.io.Sink;
9   import net.truelicense.api.io.Source;
10  import net.truelicense.api.io.Store;
11  
12  import java.io.*;
13  import java.nio.file.Path;
14  import java.util.Optional;
15  import java.util.prefs.Preferences;
16  
17  /**
18   * The standard BIOS implementation.
19   * This class is immutable.
20   *
21   * @author Christian Schlichtherle
22   */
23  public class StandardBIOS implements BIOS {
24  
25      /**
26       * {@inheritDoc}
27       * <p>
28       * The implementation in the class {@link StandardBIOS} is suitable for only
29       * small amounts of data, say a few kilobytes.
30       */
31      @Override
32      public void copy(final Source source, final Sink sink) throws IOException {
33          try (InputStream in = source.input();
34               OutputStream out = sink.output()) {
35              final byte[] buffer = new byte[Store.BUFSIZE];
36              int read;
37              while (0 <= (read = in.read(buffer)))
38                  out.write(buffer, 0, read);
39          }
40      }
41  
42      @Override
43      public Store pathStore(Path path) {
44          return new PathStore(path);
45      }
46  
47      @Override
48      public Source resource(
49              final String name,
50              final Optional<ClassLoader> classLoader) {
51          return new Source() {
52              @Override public InputStream input() throws IOException {
53                  return inputStream().orElseThrow(() -> new FileNotFoundException(name));
54              }
55  
56              Optional<InputStream> inputStream() {
57                  return classLoader
58                          .map(cl -> Optional.ofNullable(cl.getResourceAsStream(name)))
59                          .orElseGet(() -> Optional.ofNullable(ClassLoader.getSystemResourceAsStream(name)));
60              }
61          };
62      }
63  
64      @Override
65      public Source stdin() { return uncloseable(System.in); }
66  
67      @Override
68      public Sink stdout() { return uncloseable(System.out); }
69  
70      private static Source uncloseable(final InputStream in) {
71          return () -> new FilterInputStream(in) {
72              @Override public void close() { }
73          };
74      }
75  
76      private static Sink uncloseable(final OutputStream out) {
77          return () -> new FilterOutputStream(out) {
78              @Override public void close() throws IOException {
79                  out.flush();
80              }
81          };
82      }
83  
84      @Override
85      public Store systemPreferencesStore(Class<?> classInPackage, String key) {
86          return new PreferencesStore(
87                  Preferences.systemNodeForPackage(classInPackage), key);
88      }
89  
90      @Override
91      public Store userPreferencesStore(Class<?> classInPackage, String key) {
92          return new PreferencesStore(
93                  Preferences.userNodeForPackage(classInPackage), key);
94      }
95  }
