6   package net.truelicense.spi.io;
7   
8   import net.truelicense.api.codec.Codec;
9   import net.truelicense.api.io.Store;
10  import net.truelicense.spi.codec.SerializationCodec;
11  
12  import java.io.*;
13  import java.util.Optional;
14  
15  /**
16   * A (heap) memory store.
17   *
18   * @author Christian Schlichtherle
19   */
20  public final class MemoryStore implements Store {
21  
22      private final int bufsize;
23  
24      @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
25      private Optional<byte[]> optBuffer = Optional.empty();
26  
27      /**
28       * Equivalent to <code>new {@link #MemoryStore(int) MemoryStore(BUFSIZE)}</code>.
29       *
30       * @see Store#BUFSIZE
31       */
32      public MemoryStore() {
33          this(BUFSIZE);
34      }
35  
36      /**
37       * Constructs a memory store with the given data size to use upon
38       * {@link #output}.
39       *
40       * @param bufsize the data size to use upon {@link #output}.
41       */
42      public MemoryStore(final int bufsize) {
43          if (0 > (this.bufsize = bufsize)) {
44              throw new IllegalArgumentException();
45          }
46      }
47  
48      @Override
49      public InputStream input() throws IOException {
50          return new ByteArrayInputStream(checkedData());
51      }
52  
53      @Override
54      public OutputStream output() throws IOException {
55          return new ByteArrayOutputStream(bufsize) {
56              @Override
57              public void close() throws IOException {
58                  data(toByteArray());
59              }
60          };
61      }
62  
63      @Override
64      public void delete() throws IOException {
65          checkedData();
66          optBuffer = Optional.empty();
67      }
68  
69      @Override
70      public boolean exists() {
71          return optBuffer.isPresent();
72      }
73  
74      private byte[] checkedData() throws FileNotFoundException {
75          return optBuffer.orElseThrow(FileNotFoundException::new);
76      }
77  
78      public byte[] data() {
79          return optBuffer.map(byte[]::clone).orElseThrow(IllegalStateException::new);
80      }
81  
82      public MemoryStore data(final byte[] buffer) {
83          optBuffer = Optional.ofNullable(buffer.clone());
84          return this;
85      }
86  
87      /** Returns a clone of the given object using a new {@link SerializationCodec} and this memory store. */
88      public <T> T clone(T object) throws Exception { return clone(object, new SerializationCodec()); }
89  
90      /** Returns a clone of the given object using the given codec and this memory store. */
91      public <T> T clone(final T object, final Codec codec) throws Exception {
92          codec.encoder(this).encode(object);
93          return codec.decoder(this).decode(object.getClass());
94      }
95  }
