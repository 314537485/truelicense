6   package net.truelicense.spi.io;
7   
8   import net.truelicense.api.io.Sink;
9   import net.truelicense.api.io.Source;
10  import net.truelicense.api.io.Store;
11  
12  import java.io.IOException;
13  import java.nio.file.Path;
14  import java.util.Optional;
15  
16  /**
17   * A Basic Input/Output System (BIOS).
18   * Implementations need to be stateless.
19   *
20   * @author Christian Schlichtherle
21   */
22  public interface BIOS {
23  
24      /** Copies the data from the given source to the given sink. */
25      void copy(Source source, Sink sink) throws IOException;
26  
27      /** Returns a store for the given path. */
28      Store pathStore(Path path);
29  
30      /**
31       * Returns a source which loads the resource with the given {@code name}.
32       * If the given class loader list is empty, then the resource gets loaded
33       * as described in {@link ClassLoader#getSystemResourceAsStream(String)}.
34       * Otherwise, the resource gets loaded as described in
35       * {@link ClassLoader#getResourceAsStream(String)} using the first
36       * class loader in the list.
37       *
38       * @param  name the name of the resource to load.
39       * @param  classLoader
40       *         The optional class loader to use for loading the resource.
41       *         If no value is present then the system class loader gets used.
42       * @return A source which loads the resource with the given {@code name}.
43       */
44      Source resource(String name, Optional<ClassLoader> classLoader);
45  
46      /**
47       * Returns a source which reads from standard input without ever closing it.
48       */
49      Source stdin();
50  
51      /**
52       * Returns a sink which writes to standard output without ever closing it.
53       */
54      Sink stdout();
55  
56      /**
57       * Returns a store for the system preferences node for the package of the
58       * given class and the given key.
59       */
60      Store systemPreferencesStore(Class<?> classInPackage, String key);
61  
62      /**
63       * Returns a store for the user preferences node for the package of the
64       * given class and the given key.
65       */
66      Store userPreferencesStore(Class<?> classInPackage, String key);
67  }
