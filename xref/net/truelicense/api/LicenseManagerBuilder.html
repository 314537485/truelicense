6   package net.truelicense.api;
7   
8   import net.truelicense.api.auth.Authentication;
9   import net.truelicense.api.io.Transformation;
10  import net.truelicense.api.misc.Injection;
11  
12  /**
13   * A generic builder for license managers.
14   *
15   * @param <This> the specialized type for fluent programming.
16   * @author Christian Schlichtherle
17   */
18  public interface LicenseManagerBuilder<This extends LicenseManagerBuilder<This>> {
19  
20      /**
21       * Returns an injection for a key store based authentication.
22       * Call its {@link Injection#inject} method to build and inject the
23       * configured authentication into this builder and return it.
24       *
25       * @see #authentication(Authentication)
26       */
27      AuthenticationInjection<? extends This> authentication();
28  
29      /**
30       * Sets the authentication.
31       *
32       * @return {@code this}.
33       */
34      This authentication(Authentication authentication);
35  
36      /**
37       * Returns an injection for a password based encryption.
38       * Call its {@link Injection#inject} method to build and inject the
39       * configured encryption into this builder and return it.
40       *
41       * @see #encryption(Transformation)
42       */
43      EncryptionInjection<? extends This> encryption();
44  
45      /**
46       * Sets the encryption.
47       *
48       * @return {@code this}.
49       */
50      This encryption(Transformation encryption);
51  }
