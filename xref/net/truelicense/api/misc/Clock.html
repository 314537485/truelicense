6   package net.truelicense.api.misc;
7   
8   import java.util.Date;
9   
10  /**
11   * A factory for representations of the current date/time.
12   *
13   * @author Christian Schlichtherle
14   */
15  public interface Clock {
16  
17      /**
18       * Returns a new representation of the current date/time.
19       * A basic implementation could just return a new {@code Date}.
20       * However, in order to protect against date/time forgery, a more
21       * sophisticated implementation should use an authoritative time source,
22       * e.g. a radio clock, an Internet time server or extrapolate an
23       * approximation from timestamps in input data.
24       * If resolving the current date/time from an authoritative resource fails
25       * for any reason, the implementation should simply fallback to return a
26       * new {@code Date}.
27       */
28      Date now();
29  }
