6   package net.truelicense.api;
7   
8   import net.truelicense.api.codec.CodecProvider;
9   import net.truelicense.api.io.Sink;
10  import net.truelicense.api.io.Source;
11  import net.truelicense.api.io.Store;
12  import net.truelicense.api.misc.ClassLoaderProvider;
13  
14  import java.nio.file.Path;
15  
16  /**
17   * A context for license management.
18   * Use this context to configure and build a
19   * {@linkplain #vendor() vendor license manager} or a
20   * {@linkplain #consumer() consumer license manager}.
21   *
22   * @author Christian Schlichtherle
23   */
24  public interface LicenseManagementContext
25  extends ClassLoaderProvider,
26          CodecProvider,
27          LicenseFactory,
28          LicenseManagementSubjectProvider {
29  
30      /**
31       * Returns a builder for a
32       * {@linkplain ConsumerLicenseManager consumer license manager}.
33       * Call its {@link ConsumerLicenseManagerBuilder#build} method to build
34       * the configured consumer license manager.
35       */
36      ConsumerLicenseManagerBuilder consumer();
37  
38      /** Returns a new memory store. */
39      Store memoryStore();
40  
41      /** Returns a store for the given path. */
42      Store pathStore(Path path);
43  
44      /**
45       * Returns a source which loads the resource with the given {@code name}.
46       * The provided string should be computed on demand from an obfuscated form,
47       * e.g. by processing it with the TrueLicense Maven Plugin.
48       * <p>
49       * The resource will get loaded using the class loader which is returned by
50       * the method {@link #classLoader()}.
51       *
52       * @param  name the name of the resource to load.
53       * @return A source which loads the resource with the given {@code name}.
54       */
55      Source resource(String name);
56  
57      /**
58       * Returns a source which reads from standard input without ever closing it.
59       */
60      Source stdin();
61  
62      /**
63       * Returns a sink which writes to standard output without ever closing it.
64       */
65      Sink stdout();
66  
67      /**
68       * Returns a store for the system preferences node for the package of the
69       * given class.
70       * Note that the class should be excluded from byte code obfuscation or
71       * otherwise you might use an unintended store location and risk a
72       * collision with third party software.
73       * <p>
74       * The store will use the {@linkplain #subject() license subject} as its
75       * key in the preferences node.
76       */
77      Store systemPreferencesStore(Class<?> classInPackage);
78  
79      /**
80       * Returns a store for the user preferences node for the package of the
81       * given class.
82       * Note that the class should be excluded from byte code obfuscation or
83       * otherwise you might use an unintended store location and risk a
84       * collision with third party software.
85       * <p>
86       * The store will use the {@linkplain #subject() license subject} as its
87       * key in the preferences node.
88       */
89      Store userPreferencesStore(Class<?> classInPackage);
90  
91      /**
92       * Returns a builder for a
93       * {@linkplain VendorLicenseManager vendor license manager}.
94       * Call its {@link VendorLicenseManagerBuilder#build} method to build
95       * the configured vendor license manager.
96       */
97      VendorLicenseManagerBuilder vendor();
98  }
