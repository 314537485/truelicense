6   package net.truelicense.api;
7   
8   import net.truelicense.api.i18n.Message;
9   
10  import java.util.Locale;
11  
12  import static java.util.Objects.requireNonNull;
13  
14  /**
15   * Indicates that a
16   * {@linkplain LicenseValidation#validate(License) license validation} failed.
17   *
18   * @author Christian Schlichtherle
19   */
20  public class LicenseValidationException extends LicenseManagementException {
21  
22      private static final long serialVersionUID = 0L;
23  
24      private final Message msg;
25  
26      /**
27       * Constructs a license validation exception with the given
28       * internationalized message.
29       *
30       * @param msg the internationalized message.
31       */
32      public LicenseValidationException(final Message msg) {
33          this.msg = requireNonNull(msg);
34      }
35  
36      @Override
37      public String getMessage() { return msg.toString(Locale.ROOT); }
38  
39      @Override
40      public String getLocalizedMessage() {
41          return msg.toString(Locale.getDefault());
42      }
43  
44      /**
45       * Returns {@code true} if this exception is considered confidential and
46       * should not be shared with users.
47       * <p>
48       * The implementation in the class {@code LicenseValidationException}
49       * returns {@code false}.
50       */
51      @Override public boolean isConfidential() { return false; }
52  }
