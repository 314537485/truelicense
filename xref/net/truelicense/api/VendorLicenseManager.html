6   package net.truelicense.api;
7   
8   /**
9    * Defines the life cycle management operations for license keys in vendor applications alias key generators.
10   * <p>
11   * A (checked) vendor license manager generally throws a {@link LicenseManagementException} if an operation fails.
12   *
13   * @see UncheckedVendorLicenseManager#checked()
14   * @author Christian Schlichtherle
15   */
16  public interface VendorLicenseManager extends LicenseManagementSchema {
17  
18      /**
19       * Returns a license key generator for the given license bean.
20       * <p>
21       * Calling this operation performs an initial
22       * {@linkplain LicenseManagementAuthorization#clearGenerate authorization check}.
23       *
24       * @param bean the license bean to process.
25       *             This bean is not modified by the returned license key generator.
26       * @return A license key generator for the given license bean.
27       */
28      LicenseKeyGenerator generateKeyFrom(License bean) throws LicenseManagementException;
29  
30      /**
31       * Adapts this vendor license manager so that it generally throws an {@link UncheckedLicenseManagementException}
32       * instead of a (checked) {@link LicenseManagementException} if an operation fails.
33       *
34       * @return the adapted unchecked vendor license manager.
35       * @since TrueLicense 3.1.0
36       */
37      default UncheckedVendorLicenseManager unchecked() { return UncheckedLicenseManager.from(this); }
38  }
