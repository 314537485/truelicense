6   package net.truelicense.api.codec;
7   
8   import net.truelicense.api.io.Sink;
9   import net.truelicense.api.io.Source;
10  
11  /**
12   * Provides encoders and decoders for generic object graphs.
13   *
14   * @author Christian Schlichtherle
15   */
16  public interface Codec {
17  
18      /**
19       * Returns an identifier for the content type used by this codec.
20       * The returned string must conform to the syntax specified in
21       * <a href="http://tools.ietf.org/html/rfc2045">RFC2045</a>
22       * "Multipurpose Internet Mail Extensions (MIME) Part One: Format of Internet Message Bodies",
23       * <a href="http://tools.ietf.org/html/rfc2045#section-5.1">Section 5.1</a>
24       * "Syntax of the Content-Type Header Field",
25       * except that it must not start with "Content-Type:"
26       * and optional spaces, i.e. the field name must be stripped.
27       */
28      String contentType();
29  
30      /**
31       * Returns an identifier for the content transfer encoding used by this
32       * codec.
33       * The returned string must conform to the syntax specified in
34       * <a href="http://tools.ietf.org/html/rfc2045">RFC2045</a>
35       * "Multipurpose Internet Mail Extensions (MIME) Part One: Format of Internet Message Bodies",
36       * <a href="http://tools.ietf.org/html/rfc2045#section-6.1">Section 6.1</a>
37       * "Content-Transfer-Encoding Syntax",
38       * except that it must not start with "Content-Transfer-Encoding:"
39       * and optional spaces, i.e. the field name must be stripped.
40       * <p>
41       * If the Content-Transfer-Encoding equals (ignoring case) {@code "8bit"}
42       * and the  {@linkplain #contentType Content-Type} does not specify a
43       * {@code charset} parameter, then {@code UTF-8} is assumed as the charset.
44       */
45      String contentTransferEncoding();
46  
47      /** Returns an encoder which writes object graphs to the given sink. */
48      Encoder encoder(Sink sink);
49  
50      /** Returns a decoder which reads object graphs from the given source. */
51      Decoder decoder(Source source);
52  }
