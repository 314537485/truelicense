6   package net.truelicense.api.codec;
7   
8   import java.lang.reflect.Type;
9   
10  /**
11   * Decodes an object graph.
12   *
13   * @author Christian Schlichtherle
14   */
15  public interface Decoder {
16  
17      /**
18       * Decodes an object graph of the expected type.
19       *
20       * @param  <T>
21       *         the expected type of the decoded object.
22       * @param  expected
23       *         the expected type of the decoded object graph, e.g.
24       *         {@code String.class}.
25       *         This is just a hint and the implementation may ignore it.
26       * @return A duplicate of the original object graph.
27       *         Its actual type may differ from the expected type.
28       */
29      <T> T decode(Type expected) throws Exception;
30  }
