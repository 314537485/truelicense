6   package net.truelicense.api;
7   
8   import net.truelicense.api.io.Source;
9   
10  /**
11   * Defines the life cycle management operations for license keys in consumer applications.
12   * <p>
13   * A (checked) consumer license manager generally throws a {@link LicenseManagementException} if an operation fails.
14   *
15   * <h3>How to Preview License Keys</h3>
16   * <p>
17   * This interface intentionally lacks a method to preview the license bean which is encoded in a license key.
18   * You can work around this constraint by using a {@linkplain LicenseManagementContext license management context} to
19   * configure a consumer license manager which uses a transient
20   * {@linkplain LicenseManagementContext#memoryStore() memory store} instead of a persistent store - see
21   * {@link ConsumerLicenseManagerBuilder#storeIn}.
22   * Once configured, you can {@linkplain #install install} the license key to the transient memory store and
23   * {@linkplain #load load} its encoded license bean.
24   *
25   * @see UncheckedConsumerLicenseManager#checked()
26   * @author Christian Schlichtherle
27   */
28  public interface ConsumerLicenseManager extends LicenseManagementSchema {
29  
30      /**
31       * Verifies the digital signature of the license key and copies it to the
32       * configured store.
33       * Unlike {@link #verify}, this operation does <em>not</em> validate the
34       * encoded license bean.
35       * This enables the caller to obtain a duplicate of the license bean even
36       * if its validation would fail, e.g. if the license has expired.
37       * <p>
38       * Calling this operation performs an initial
39       * {@linkplain LicenseManagementAuthorization#clearInstall authorization check}.
40       *
41       * @param  source the source for loading the license key.
42       */
43      void install(Source source) throws LicenseManagementException;
44  
45      /**
46       * Eventually loads the installed license key and returns an unvalidated
47       * duplicate of its encoded license bean.
48       * Unlike {@link #verify}, this operation does <em>not</em> validate the
49       * encoded license bean.
50       * This enables the caller to obtain a duplicate of the license bean even
51       * if its validation would fail, e.g. if the license has expired.
52       * <p>
53       * Calling this operation performs an initial
54       * {@linkplain LicenseManagementAuthorization#clearLoad authorization check}.
55       *
56       * @return An unvalidated duplicate of the license bean which is encoded in
57       *         the installed license key.
58       */
59      License load() throws LicenseManagementException;
60  
61      /**
62       * Eventually loads the installed license key and verifies its encoded
63       * license bean.
64       * You should call this method whenever you want to verify access to a
65       * feature of your software product.
66       * Execution needs to be fast in order to support frequent calling.
67       * <p>
68       * Calling this operation performs an initial
69       * {@linkplain LicenseManagementAuthorization#clearVerify authorization check}.
70       *
71       * @throws LicenseValidationException if validating the license bean fails,
72       *         e.g. if the license has expired.
73       */
74      void verify() throws LicenseManagementException;
75  
76      /**
77       * Uninstalls the installed license key.
78       * <p>
79       * Calling this operation performs an initial
80       * {@linkplain LicenseManagementAuthorization#clearUninstall authorization check}.
81       */
82      void uninstall() throws LicenseManagementException;
83  
84      /**
85       * Adapts this consumer license manager so that it generally throws an {@link UncheckedLicenseManagementException}
86       * instead of a (checked) {@link LicenseManagementException} if an operation fails.
87       *
88       * @return the adapted unchecked consumer license manager.
89       * @since TrueLicense 3.1.0
90       */
91      default UncheckedConsumerLicenseManager unchecked() { return UncheckedLicenseManager.from(this); }
92  }
