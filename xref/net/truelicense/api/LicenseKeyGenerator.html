6   package net.truelicense.api;
7   
8   import net.truelicense.api.io.Sink;
9   
10  /**
11   * Generates a license key and writes it to a given sink or returns a duplicate of its encoded license bean.
12   * License key generators are the product of a call to {@link VendorLicenseManager#generateKeyFrom(License)}.
13   * License key generators are stateful and so they are generally <em>not</em> thread-safe.
14   * <p>
15   * When the license key gets generated, a protective copy of the configured license bean is made which is subsequently
16   * {@linkplain LicenseInitialization#initialize initialized} and {@linkplain LicenseValidation#validate validated}.
17   *
18   * @author Christian Schlichtherle
19   */
20  public interface LicenseKeyGenerator {
21  
22      /** Returns a duplicate of the license bean which is encoded in the generated license key. */
23      License license() throws LicenseManagementException;
24  
25      /**
26       * Saves the generated license key to the given sink.
27       *
28       * @param sink the sink for writing the generated license key to.
29       * @return {@code this}
30       */
31      LicenseKeyGenerator saveTo(Sink sink) throws LicenseManagementException;
32  }
