6   package net.truelicense.ui.wizard;
7   
8   /**
9    * A basic implementation of a wizard controller.
10   * This class is immutable.
11   *
12   * @param  <S> the type of the wizard's states.
13   * @param  <V> the type of the wizard's views.
14   * @since  TrueLicense 2.3
15   * @author Christian Schlichtherle
16   */
17  public abstract class BasicWizardController<S, V extends WizardView<S>>
18  implements WizardController {
19  
20      /**
21       * Creates a new wizard controller with the given model.
22       *
23       * @param model the wizard model.
24       * @param <S> the type of the wizard's states.
25       * @param <V> the type of the wizard's views.
26       */
27      public static <S, V extends WizardView<S>> WizardController create(
28              final WizardModel<S, V> model) {
29          return new BasicWizardController<S, V>() {
30              @Override protected WizardModel<S, V> model() { return model; }
31          };
32      }
33  
34      /** Returns the wizard model. */
35      protected abstract WizardModel<S, V> model();
36  
37      @Override public final boolean switchBackEnabled() {
38          return !backState().equals(currentState());
39      }
40  
41      @Override public final void switchBack() { switchTo(backState()); }
42  
43      @Override public final boolean switchNextEnabled() {
44          return !nextState().equals(currentState());
45      }
46  
47      @Override public final void switchNext() { switchTo(nextState()); }
48  
49      /** Switches from the current state to the given state. */
50      private void switchTo(final S newState) {
51          if (newState.equals(currentState())) return;
52          fireBeforeStateSwitch();
53          currentState(newState);
54          fireAfterStateSwitch();
55      }
56  
57      /**
58       * Notifies the current view of the event by calling its
59       * {@link WizardView#onBeforeStateSwitch} method and then calls this
60       * controller's {@link #onBeforeStateSwitch} hook.
61       * This method gets called before this controller switches the state of
62       * the model.
63       */
64      protected final void fireBeforeStateSwitch() {
65          currentView().onBeforeStateSwitch();
66          onBeforeStateSwitch();
67      }
68  
69      /**
70       * Override this template method in order to add some behaviour before this
71       * controller switches the state of the model, but after the current view
72       * has been notified via {@link WizardView#onBeforeStateSwitch}.
73       *
74       * @see #currentView
75       */
76      protected void onBeforeStateSwitch() { }
77  
78      /**
79       * Calls this controller's {@link #onAfterStateSwitch} hook and then
80       * notifies the current view of the event by calling its
81       * {@link WizardView#onAfterStateSwitch} method.
82       * This method gets called after this controller has switched the state of
83       * the model.
84       */
85      protected final void fireAfterStateSwitch() {
86          onAfterStateSwitch();
87          currentView().onAfterStateSwitch();
88      }
89  
90      /**
91       * Override this template method in order to add some behaviour after this
92       * controller has switched the state of the model, but before the current
93       * view gets notified via {@link WizardView#onAfterStateSwitch}.
94       *
95       * @see #currentView
96       */
97      protected void onAfterStateSwitch() { }
98  
99      /** Returns the view of the current state. */
100     protected final V currentView() { return view(currentState()); }
101 
102     /** Returns the view for the given state. */
103     protected final V view(S state) { return model().view(state); }
104 
105     /** Sets the view for the given state. */
106     protected void view(S state, V view) { model().view(state, view); }
107 
108     /** Returns the current state. */
109     protected final S currentState() { return model().currentState(); }
110 
111     /** Sets the current state. */
112     private void currentState(S state) { model().currentState(state); }
113 
114     /**
115      * Returns the state to switch to when the back-button gets pressed,
116      * or the current state if switching is not possible.
117      */
118     protected final S backState() { return currentView().backState(); }
119 
120     /**
121      * Returns the state to switch to when the next-button gets pressed,
122      * or the current state if switching is not possible.
123      */
124     protected final S nextState() { return currentView().nextState(); }
125 }
