6   package net.truelicense.maven.plugin.obfuscation;
7   
8   import net.truelicense.maven.plugin.commons.MojoAdapter;
9   import net.truelicense.maven.plugin.obfuscation.core.Processor;
10  import net.truelicense.obfuscate.ObfuscatedString;
11  import org.apache.maven.plugin.MojoFailureException;
12  import org.apache.maven.plugins.annotations.Parameter;
13  
14  import java.io.File;
15  import java.io.IOException;
16  import java.util.Formatter;
17  
18  /**
19   * A basic MOJO for the obfuscation of constant string values in Java class
20   * files (byte code).
21   *
22   * @see ObfuscatedString
23   * @since TrueLicense 2.4
24   * @author Christian Schlichtherle
25   */
26  public abstract class ObfuscateClassesMojo extends MojoAdapter {
27  
28      @Parameter(defaultValue = "${project.build.directory}", readonly = true)
29      private File buildDirectory;
30  
31      /** The maximum allowed size of a class file in bytes. */
32      @Parameter(property = "truelicense.obfuscate.maxBytes", defaultValue = "" + Processor.DEFAULT_MAX_BYTES)
33      private int maxBytes;
34  
35      /**
36       * The scope of constant string value obfuscation:
37       * <ul>
38       * <li>Use <code>none</code> to skip obfuscation.
39       * <li>Use <code>annotated</code> to obfuscate only constant string
40       *     values of fields which have been annotated with @Obfuscate.
41       * <li>Use <code>all</code> to obfuscate all constant string values.
42       *     This may result in quite some code size and runtime overhead,
43       *     so you should <em>not</em> use this in general.
44       * </ul>
45       */
46      @Parameter(property = "truelicense.obfuscate.scope", defaultValue = "annotated")
47      private Scope scope;
48  
49      /**
50       * The format for synthesized method names.
51       * This a format string for the class {@link Formatter}.
52       * It's first parameter is a string identifier for the obfuscation stage
53       * and its second parameter is an integer index for the synthesized method.
54       */
55      @Parameter(property = "truelicense.obfuscate.methodNameFormat", defaultValue = Processor.DEFAULT_METHOD_NAME_FORMAT)
56      private String methodNameFormat;
57  
58      /**
59       * Whether or not a call to <code>java.lang.String.intern()</code>
60       * shall get added when computing the original constant string values again.
61       * Use this to preserve the identity relation of constant string values if
62       * required.
63       */
64      @Parameter(property = "truelicense.obfuscate.intern", defaultValue = "" + Processor.DEFAULT_INTERN_STRINGS)
65      private boolean intern;
66  
67      protected abstract File outputDirectory();
68  
69      @Override
70      protected final void doExecute() throws MojoFailureException {
71          try {
72              if (scope == Scope.none) {
73                  getLog().warn("Skipping constant string value obfuscation.");
74              } else {
75                  getLog().info(String.format(
76                          "Obfuscating %s constant string values in %s.",
77                          scope,
78                          outputDirectory()));
79                  obfuscate();
80              }
81          } catch (IOException e) {
82              throw new MojoFailureException(e.toString(), e);
83          }
84      }
85  
86      private void obfuscate() throws IOException {
87          Processor.builder()
88                  .logger(new LoggerAdapter(getLog()))
89                  .directory(outputDirectory().toPath())
90                  .maxBytes(maxBytes)
91                  .obfuscateAll(scope == Scope.all)
92                  .methodNameFormat(methodNameFormat)
93                  .internStrings(intern)
94                  .build()
95                  .execute();
96      }
97  }
