6   package net.truelicense.maven.plugin.obfuscation.core;
7   
8   import org.objectweb.asm.ClassReader;
9   import org.objectweb.asm.ClassWriter;
10  
11  import java.io.IOException;
12  
13  import static org.objectweb.asm.ClassWriter.COMPUTE_MAXS;
14  
15  /**
16   * @author Christian Schlichtherle
17   */
18  final class SecondPass extends Pass {
19  
20      /**
21       * This (virtually) unique prefix is required to make the processor
22       * idempotent.
23       */
24      private final String prefix = "clinit@" + System.currentTimeMillis();
25  
26      SecondPass(Processor ctx) { super(ctx); }
27  
28      @Override
29      public void execute() throws IOException {
30          logger().debug("Running second pass.");
31          super.execute();
32      }
33  
34      @Override
35      void process(final Node node) throws IOException {
36          logger(node).trace("Transforming class file.");
37          final ClassReader cr = new ClassReader(read(node));
38          final ClassWriter cw = new ClassWriter(COMPUTE_MAXS);
39          cr.accept(ctx.obfuscator(ctx.merger(cw, prefix)), 0);
40          write(node, cw.toByteArray());
41      }
42  }
