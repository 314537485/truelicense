6   package net.truelicense.maven.plugin.obfuscation.core;
7   
8   import java.nio.file.Path;
9   
10  /**
11   * @author Christian Schlichtherle
12   */
13  final class Node {
14  
15      private final Path base, file;
16  
17      Node(final Path base) {
18          this.base = this.file = base;
19      }
20  
21      private Node(final Path base, final Path resolvedMember) {
22          this.base = base;
23          this.file = resolvedMember;
24      }
25  
26      Node updateFile(final Path resolvedFile) {
27          assert base.toString().isEmpty() && !resolvedFile.isAbsolute() || resolvedFile.startsWith(base);
28          return new Node(base, resolvedFile);
29      }
30  
31      Path file() { return file; }
32  
33      String path() { return base.relativize(file).toString(); }
34  }
