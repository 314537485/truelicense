6   package net.truelicense.maven.plugin.obfuscation.core;
7   
8   import org.objectweb.asm.ClassReader;
9   import org.slf4j.Logger;
10  
11  import java.io.IOException;
12  
13  import static org.objectweb.asm.ClassReader.*;
14  
15  /**
16   * @author Christian Schlichtherle
17   */
18  final class FirstPass extends Pass {
19  
20      private static final int
21              CLASS_READER_FLAGS = SKIP_DEBUG | SKIP_FRAMES | SKIP_CODE;
22  
23      FirstPass(Processor ctx) { super(ctx); }
24  
25      @Override
26      public void execute() throws IOException {
27          final Logger logger = logger();
28          if (obfuscateAll()) {
29              logger.debug("Skipping first pass because all constant string values should get obfuscated.");
30          } else {
31              logger.debug("Running first pass.");
32              super.execute();
33          }
34      }
35  
36      @Override
37      void process(final Node node) throws IOException {
38          logger(node).trace("Analyzing class file.");
39          new ClassReader(read(node)).accept(ctx.collector(), CLASS_READER_FLAGS);
40      }
41  }
