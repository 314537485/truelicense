6   package net.truelicense.v2.xml;
7   
8   import net.truelicense.api.License;
9   import net.truelicense.api.codec.Codec;
10  import net.truelicense.v2.commons.V2LicenseApplicationContext;
11  import net.truelicense.v2.commons.auth.V2RepositoryModel;
12  import net.truelicense.v2.xml.codec.JaxbCodec;
13  
14  import javax.xml.bind.JAXBContext;
15  import javax.xml.bind.JAXBException;
16  
17  /**
18   * The root context for the management of Version-2-with-XML (V2/XML) format
19   * license keys.
20   * Note that there is no compatibility between different format license keys.
21   * <p>
22   * This class is immutable.
23   *
24   * @author Christian Schlichtherle
25   */
26  public class V2XmlLicenseApplicationContext
27  extends V2LicenseApplicationContext {
28  
29      private volatile JAXBContext context;
30  
31      /**
32       * {@inheritDoc}
33       * <p>
34       * The implementation in the class {@link V2XmlLicenseApplicationContext}
35       * returns a {@link JaxbCodec}.
36       */
37      @Override
38      public Codec codec() { return new JaxbCodec(jaxbContext()); }
39  
40      private JAXBContext jaxbContext() {
41          final JAXBContext c = context;
42          return null != c ? c : (context = newJaxbContext());
43      }
44  
45      /**
46       * Returns a new JAXB context for use with {@linkplain #license licenses}
47       * and {@linkplain #repositoryContext repositories}.
48       * This method is normally only called once.
49       * In a multi-threaded environment, it may get called more than once, but
50       * then each invocation must return an object which behaves equivalent to
51       * any previously returned object.
52       * <p>
53       * The implementation in the class {@link V2XmlLicenseApplicationContext}
54       * constructs a new {@code JAXBContext} for the root element classes
55       * {@link License} and {@link V2RepositoryModel}.
56       */
57      protected JAXBContext newJaxbContext() {
58          try {
59              return JAXBContext.newInstance(License.class, V2RepositoryModel.class);
60          } catch (final JAXBException ex) {
61              throw new AssertionError(ex);
62          }
63      }
64  }
