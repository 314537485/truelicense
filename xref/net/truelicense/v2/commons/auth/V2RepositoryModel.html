6   package net.truelicense.v2.commons.auth;
7   
8   import net.truelicense.core.misc.Strings;
9   
10  import javax.xml.bind.annotation.XmlElement;
11  import javax.xml.bind.annotation.XmlRootElement;
12  import javax.xml.bind.annotation.XmlType;
13  import java.util.Objects;
14  
15  import static java.util.Locale.ENGLISH;
16  
17  /**
18   * A repository model for use with V2 format license keys.
19   * All properties are set to {@code null} by default.
20   *
21   * @author Christian Schlichtherle
22   */
23  // #TRUELICENSE-50: The XML element name MUST be explicitly defined!
24  // Otherwise, it would get derived from the class name, but this would break
25  // if the class name gets obfuscated, e.g. when using the ProGuard
26  // configuration from the TrueLicense Maven Archetype.
27  @XmlRootElement(name = "repository")
28  // #TRUELICENSE-52: Dito for the XML type. This enables objects of this class
29  // to participate in larger object graphs which the application wants to
30  // encode/decode to/from XML.
31  @XmlType(name = "repository")
32  public final class V2RepositoryModel {
33  
34      private String algorithm, artifact, signature;
35  
36      /** Returns the signature algorithm. */
37      @XmlElement(required = true)
38      public final String getAlgorithm() { return algorithm; }
39  
40      /** Sets the signature algorithm. */
41      public final void setAlgorithm(final String algorithm) {
42          this.algorithm = algorithm;
43      }
44  
45      /** Returns the encoded artifact. */
46      @XmlElement(required = true)
47      public final String getArtifact() { return artifact; }
48  
49      /** Sets the encoded artifact. */
50      public final void setArtifact(final String artifact) {
51          this.artifact = artifact;
52      }
53  
54      /** Returns the encoded signature. */
55      @XmlElement(required = true)
56      public final String getSignature() { return signature; }
57  
58      /** Sets the encoded signature. */
59      public void setSignature(final String signature) {
60          this.signature = signature;
61      }
62  
63      /**
64       * Returns {@code true} if and only if the given object is an instance of
65       * {@link V2RepositoryModel} and it's signature algorithm, encoded
66       * artifact and encoded signature compare equal, whereby the case of the
67       * signature algorithm is ignored.
68       */
69      @Override
70      public boolean equals(final Object obj) {
71          if (this == obj) return true;
72          if (!(obj instanceof V2RepositoryModel)) return false;
73          final V2RepositoryModel that = (V2RepositoryModel) obj;
74          return  Objects.equals(this.getArtifact(), that.getArtifact()) &&
75                  Objects.equals(this.getSignature(), that.getSignature()) &&
76                  Strings.equalsIgnoreCase(this.getAlgorithm(), that.getAlgorithm());
77      }
78  
79      /** Returns a hash code which is consistent with {@link #equals}. */
80      @Override
81      public int hashCode() {
82          int c = 17;
83          c = 31 * c + Objects.hashCode(getArtifact());
84          c = 31 * c + Objects.hashCode(getSignature());
85          c = 31 * c + Objects.hashCode(Strings.toLowerCase(getAlgorithm(), ENGLISH));
86          return c;
87      }
88  }
