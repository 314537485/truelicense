6   package net.truelicense.jax.rs;
7   
8   import net.truelicense.api.LicenseManagementException;
9   
10  import javax.ws.rs.core.Response.StatusType;
11  import java.security.GeneralSecurityException;
12  
13  /**
14   * Wraps a {@link LicenseManagementException} in order to decorate it with
15   * additional meta data for generating an HTTP response.
16   * This class is immutable.
17   *
18   * @since  TrueLicense 2.3
19   * @author Christian Schlichtherle
20   */
21  public final class ConsumerLicenseManagementServiceException
22  extends GeneralSecurityException {
23  
24      private static final long serialVersionUID = 0L;
25  
26      private final StatusType status;
27  
28      /**
29       * Constructs a consumer license management service exception with the given HTTP
30       * status code and its causing license management exception.
31       *
32       * @param status the HTTP status code.
33       * @param nullableCause the nullable causing license management exception.
34       * @see <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html">http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html</a>
35       */
36      ConsumerLicenseManagementServiceException(
37              final StatusType status,
38              final LicenseManagementException nullableCause) {
39          super(null == nullableCause ? null : nullableCause.getMessage(), nullableCause);
40          this.status = status;
41      }
42  
43      /** Returns the HTTP status type. */
44      public StatusType getStatus() { return status; }
45  }
