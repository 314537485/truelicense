6   package net.truelicense.swing;
7   
8   import net.truelicense.api.ConsumerLicenseManager;
9   import net.truelicense.swing.util.Enabler;
10  
11  import javax.swing.*;
12  import java.io.Serializable;
13  import java.lang.reflect.InvocationTargetException;
14  
15  /**
16   * A decorating consumer license manager which hosts an {@link Enabler}.
17   * This class is immutable.
18   *
19   * @author Christian Schlichtherle
20   */
21  abstract class UpdatingConsumerLicenseManager
22  extends DecoratingConsumerLicenseManager
23  implements Serializable {
24  
25      private final Enabler enabler;
26  
27      UpdatingConsumerLicenseManager(final ConsumerLicenseManager manager, final Enabler enabler) {
28          super(manager);
29          assert null != enabler;
30          this.enabler = enabler;
31      }
32  
33      final void enable() { enabled(true); }
34      final void disable() { enabled(false); }
35  
36      final boolean enabled() {
37          class Action implements Runnable {
38              @SuppressWarnings("WeakerAccess")
39              boolean result;
40  
41              @Override public void run() { result = enabler.enabled(); }
42          }
43          return runOnEventDispatchThread(new Action()).result;
44      }
45  
46      final void enabled(final boolean value) { runOnEventDispatchThread(() -> enabler.enabled(value)); }
47  
48      private <R extends Runnable> R runOnEventDispatchThread(R action) {
49          if (SwingUtilities.isEventDispatchThread()) {
50              action.run();
51          } else {
52              try {
53                  SwingUtilities.invokeAndWait(action);
54              } catch (InterruptedException ex) {
55                  action.run(); // never mind!
56              } catch (InvocationTargetException ex) {
57                  throw new AssertionError(ex);
58              }
59          }
60          return action;
61      }
62  }
