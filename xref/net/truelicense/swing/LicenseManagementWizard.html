6   package net.truelicense.swing;
7   
8   import net.truelicense.api.ConsumerLicenseManager;
9   import net.truelicense.swing.util.EnhancedDialog;
10  import net.truelicense.swing.wizard.SwingWizardController;
11  import net.truelicense.swing.util.ComponentEnabler;
12  import net.truelicense.ui.LicenseWizardMessage;
13  import net.truelicense.ui.LicenseWizardState;
14  import net.truelicense.ui.wizard.BasicWizardModel;
15  import net.truelicense.ui.wizard.WizardModel;
16  
17  import java.awt.*;
18  
19  import static net.truelicense.ui.LicenseWizardState.*;
20  
21  /**
22   * An internationalized wizard dialog for license management in consumer
23   * applications.
24   *
25   * @author Christian Schlichtherle
26   */
27  public final class LicenseManagementWizard {
28  
29      /** Indicates that the "Finish" component was pressed to close the dialog. */
30      public static final int FINISH_RETURN_CODE =
31              SwingWizardController.ReturnCode.finish.ordinal();
32  
33      /** Indicates that the "Cancel" component was pressed to close the dialog. */
34      public static final int CANCEL_RETURN_CODE =
35              SwingWizardController.ReturnCode.cancel.ordinal();
36  
37      private final LicenseWizardController controller;
38      private final ConsumerLicenseManager manager;
39  
40      /**
41       * Constructs a license wizard.
42       *
43       * @param manager the consumer license manager.
44       */
45      public LicenseManagementWizard(ConsumerLicenseManager manager) {
46          this(manager, (Frame) null);
47      }
48  
49      /**
50       * Constructs a license wizard for the given owner dialog.
51       *
52       * @param manager the consumer license manager.
53       * @param owner the owner dialog.
54       */
55      public LicenseManagementWizard(
56              ConsumerLicenseManager manager,
57              Dialog owner) {
58          this(new EnhancedDialog(owner), manager);
59      }
60  
61      /**
62       * Constructs a license wizard for the given owner frame.
63       *
64       * @param manager the consumer license manager.
65       * @param owner the owner frame.
66       */
67      public LicenseManagementWizard(
68              ConsumerLicenseManager manager,
69              Frame owner) {
70          this(new EnhancedDialog(owner), manager);
71      }
72  
73      @SuppressWarnings("LeakingThisInConstructor")
74      private LicenseManagementWizard(
75              final EnhancedDialog dialog,
76              final ConsumerLicenseManager manager) {
77          dialog.setTitle(LicenseWizardMessage.wizard_title
78                  .format(manager.context().subject()).toString());
79          this.manager = manager;
80          this.controller = new LicenseWizardController(dialog);
81          this.controller.setupPanels(this);
82      }
83  
84      void enableNextButton() { nextButtonProxy().enable(); }
85      void disableNextButton() { nextButtonProxy().disable(); }
86  
87      ComponentEnabler nextButtonProxy() { return controller.nextButtonProxy(); }
88  
89      ConsumerLicenseManager manager() { return manager; }
90  
91      /**
92       * Returns {@code true} if and only if the uninstall-button is visible on
93       * the welcome panel.
94       * By default, this button is <em>not</em> visible.
95       */
96      public boolean isUninstallButtonVisible() {
97          return controller.isUninstallButtonVisible();
98      }
99  
100     /** Sets the visibility of the uninstall-button. */
101     public void setUninstallButtonVisible(boolean visible) {
102         controller.setUninstallButtonVisible(visible);
103     }
104 
105     /**
106      * Packs and displays the modal wizard dialog.
107      *
108      * @return An integer that identifies how the dialog was closed.
109      *         See return code constants at the beginning of this class.
110      */
111     public int showModalDialog() {
112         return controller.showModalDialog().ordinal();
113     }
114 
115     /**
116      * Retrieves the last return code set by the dialog.
117      *
118      * @return An integer that identifies how the dialog was closed.
119      *         See RETURN_CODE constants at the beginning of this class.
120      */
121     public int getReturnCode() { return controller.lastReturnCode().ordinal(); }
122 
123     private static final class LicenseWizardController
124     extends SwingWizardController<LicenseWizardState, LicensePanel> {
125 
126         private final WizardModel<LicenseWizardState, LicensePanel>
127                 model = BasicWizardModel.create(LicenseWizardState.class);
128 
129         LicenseWizardController(EnhancedDialog dialog) { super(dialog); }
130 
131         @Override protected WizardModel<LicenseWizardState, LicensePanel> model() {
132             return model;
133         }
134 
135         @Override protected ComponentEnabler nextButtonProxy() {
136             return super.nextButtonProxy();
137         }
138 
139         void setupPanels(final LicenseManagementWizard wizard) {
140             view(welcome, new WelcomePanel(wizard));
141             view(install, new InstallPanel(wizard));
142             view(display, new DisplayPanel(wizard));
143             view(uninstall, new UninstallPanel(wizard));
144         }
145 
146         boolean isUninstallButtonVisible() {
147             return welcomePanel().isUninstallButtonVisible();
148         }
149 
150         void setUninstallButtonVisible(boolean visible) {
151             welcomePanel().setUninstallButtonVisible(visible);
152         }
153 
154         private WelcomePanel welcomePanel() {
155             return (WelcomePanel) super.view(LicenseWizardState.welcome);
156         }
157     }
158 }
