6   package net.truelicense.swing;
7   
8   import net.truelicense.api.ConsumerLicenseManager;
9   import net.truelicense.api.LicenseManagementException;
10  import net.truelicense.api.io.Source;
11  import net.truelicense.swing.util.Enabler;
12  
13  /**
14   * A decorating consumer license manager which disables a component before it
15   * forwards the call to {@link #install} or {@link #uninstall} to the delegate
16   * manager.
17   * If the operation succeeds, the component remains disabled.
18   * Otherwise, the component state gets recovered.
19   * This class is immutable.
20   *
21   * @author Christian Schlichtherle
22   */
23  final class DisablingConsumerLicenseManager
24  extends UpdatingConsumerLicenseManager {
25  
26      private static final long serialVersionUID = 0L;
27  
28      DisablingConsumerLicenseManager(Enabler enabler, ConsumerLicenseManager manager) { super(manager, enabler); }
29  
30      @Override
31      public void install(final Source source) throws LicenseManagementException { run(() -> manager.install(source)); }
32  
33      @Override
34      public void uninstall() throws LicenseManagementException { run(manager::uninstall); }
35  
36      private void run(final CheckedTask task) throws LicenseManagementException {
37          final boolean enabled = enabled();
38          disable();
39          try {
40              task.run();
41          } catch (final Throwable e) {
42              enabled(enabled);
43              throw e;
44          }
45      }
46  
47      private interface CheckedTask { void run() throws LicenseManagementException; }
48  }
