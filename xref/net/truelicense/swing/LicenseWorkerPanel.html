6   package net.truelicense.swing;
7   
8   import net.truelicense.api.LicenseManagementException;
9   import net.truelicense.api.i18n.Message;
10  import net.truelicense.spi.i18n.BasicMessage;
11  import net.truelicense.ui.LicenseWizardMessage;
12  
13  import javax.swing.*;
14  import java.util.Locale;
15  import java.util.concurrent.ExecutionException;
16  
17  import static javax.swing.SwingUtilities.isEventDispatchThread;
18  
19  /**
20   * An abstract {@code JPanel} for license consumer management.
21   *
22   * @author Christian Schlichtherle
23   */
24  abstract class LicenseWorkerPanel extends LicensePanel {
25  
26      static final Message EMPTY_MESSAGE = new BasicMessage() {
27          static final long serialVersionUID = 0L;
28  
29          @Override public String toString(Locale locale) { return ""; }
30      };
31  
32      LicenseWorkerPanel(LicenseManagementWizard wizard) { super(wizard); }
33  
34      abstract void setStatusMessage(Message message);
35  
36      abstract Message successMessage();
37      abstract Message failureMessage(Throwable throwable);
38  
39      static boolean isConsideredConfidential(Throwable ex) {
40          return ex instanceof LicenseManagementException &&
41                  ((LicenseManagementException) ex).isConfidential();
42      }
43  
44      @SuppressWarnings("PackageVisibleInnerClass")
45      abstract class LicenseWorker extends SwingWorker<Void, Void> {
46  
47          @Override protected final void done() {
48              try {
49                  get();
50                  setStatusMessage(successMessage());
51              } catch (ExecutionException ex) {
52                  showExceptionDialog(ex.getCause());
53              } catch (InterruptedException ex) {
54                  showExceptionDialog(ex);
55              }
56  
57          }
58  
59          void showExceptionDialog(final Throwable ex) {
60              assert isEventDispatchThread();
61              JOptionPane.showMessageDialog(
62                      LicenseWorkerPanel.this,
63                      isConsideredConfidential(ex)
64                          ? failureMessage(ex).toString()
65                          : ex.getLocalizedMessage(),
66                      LicenseWizardMessage.failure_title.format().toString(),
67                      JOptionPane.ERROR_MESSAGE);
68  
69          }
70      }
71  }
