6   package net.truelicense.swing.wizard;
7   
8   import net.truelicense.swing.util.ComponentEnabler;
9   import net.truelicense.swing.util.EnhancedButton;
10  import net.truelicense.swing.util.EnhancedDialog;
11  import net.truelicense.ui.wizard.BasicWizardController;
12  import net.truelicense.ui.wizard.WizardMessage;
13  import net.truelicense.ui.wizard.WizardView;
14  
15  import javax.swing.*;
16  import javax.swing.border.Border;
17  import javax.swing.border.EmptyBorder;
18  import java.awt.*;
19  import java.awt.event.ActionListener;
20  import java.util.Objects;
21  
22  import static net.truelicense.ui.wizard.WizardMessage.*;
23  
24  /**
25   * A generic wizard which uses a {@link JDialog}.
26   *
27   * @param <S> the type of the wizard's states.
28   * @param <V> the type of the wizard's views.
29   * @since  TrueLicense 2.3
30   * @author Christian Schlichtherle
31   */
32  public abstract class SwingWizardController<
33          S extends Enum<S>,
34          V extends Component & WizardView<S>>
35  extends BasicWizardController<S, V> {
36  
37      /** Indicates which button was pressed to close the wizard dialog. */
38      // Don't change order: The ordinal may be used by clients!
39      public enum ReturnCode { finish, cancel }
40  
41      private final JDialog dialog;
42      private final JButton cancelButton, backButton, nextButton;
43      private final CardLayout cardLayout;
44      private final JPanel cardPanel;
45  
46      private ReturnCode lastReturnCode = ReturnCode.cancel;
47  
48      /**
49       * Constructs a wizard controller for the given owner dialog.
50       *
51       * @param nullableOwner the nullable owner dialog.
52       */
53      protected SwingWizardController(Dialog nullableOwner) {
54          this(new EnhancedDialog(nullableOwner));
55      }
56  
57      /**
58       * Constructs a wizard controller for the given owner frame.
59       *
60       * @param nullableOwner the nullable owner frame.
61       */
62      protected SwingWizardController(Frame nullableOwner) {
63          this(new EnhancedDialog(nullableOwner));
64      }
65  
66      /**
67       * Constructs a wizard controller which uses the given dialog.
68       *
69       * @param dialog the dialog to use.
70       */
71      protected SwingWizardController(final EnhancedDialog dialog) {
72          final ActionListener listener = e -> {
73              final WizardMessage key = WizardMessage.valueOf(e.getActionCommand());
74              if      (wizard_cancel.equals(key)) cancel();
75              else if (wizard_back.equals(key))   switchBack();
76              else if (wizard_next.equals(key))   switchNext();
77              else if (wizard_finish.equals(key)) finish();
78              else                                throw new AssertionError();
79          };
80  
81          cancelButton = new EnhancedButton();
82          cancelButton.setName(wizard_cancel.name());
83          cancelButton.setActionCommand(wizard_cancel.name());
84          cancelButton.addActionListener(listener);
85  
86          backButton = new EnhancedButton();
87          backButton.setName(wizard_back.name());
88          backButton.setActionCommand(wizard_back.name());
89          backButton.addActionListener(listener);
90  
91          nextButton = new EnhancedButton();
92          nextButton.setName(wizard_next.name());
93          nextButton.setActionCommand(wizard_next.name());
94          nextButton.addActionListener(listener);
95  
96          final Border border = new EmptyBorder(new Insets(10, 10, 10, 10));
97  
98          final Box buttonBox = new Box(BoxLayout.LINE_AXIS);
99          buttonBox.setBorder(border);
100         buttonBox.add(backButton);
101         buttonBox.add(Box.createHorizontalStrut(10));
102         buttonBox.add(nextButton);
103         buttonBox.add(Box.createHorizontalStrut(30));
104         buttonBox.add(cancelButton);
105 
106         //  Create the buttons with a separator above them, then place them
107         //  on the east side of the view with a small amount of space between
108         //  the back and the next button, and a larger amount of space between
109         //  the next button and the cancel button.
110         final JPanel controlPanel = new JPanel(new BorderLayout());
111         controlPanel.add(new JSeparator(), BorderLayout.NORTH);
112         controlPanel.add(buttonBox, BorderLayout.EAST);
113 
114         cardLayout = new CardLayout();
115 
116         cardPanel = new JPanel(cardLayout);
117         cardPanel.setBorder(border);
118 
119         this.dialog = Objects.requireNonNull(dialog);
120         dialog.setName(getClass().getSimpleName());
121         dialog.setLayout(new BorderLayout());
122         dialog.add(cardPanel, BorderLayout.CENTER);
123         dialog.add(controlPanel, BorderLayout.SOUTH);
124         dialog.getRootPane().setDefaultButton(nextButton);
125     }
126 
127     @Override protected void view(S state, V view) {
128         super.view(state, view);
129         cardPanel.add(view, state.name());
130     }
131 
132     /** Returns an enabler for the next-button. */
133     protected ComponentEnabler nextButtonProxy() {
134         return new ComponentEnabler() {
135             static final long serialVersionUID = 0L;
136 
137             @Override protected JButton component() { return nextButton; }
138 
139             @Override public void enabled(final boolean value) {
140                 assert SwingUtilities.isEventDispatchThread();
141                 super.enabled(value);
142                 requestFocusOnBackButtonIfEnabled();
143                 requestFocusOnNextButtonIfEnabled();
144             }
145         };
146     }
147 
148     /** Packs and displays the modal wizard dialog. */
149     public ReturnCode showModalDialog() {
150         fireAfterStateSwitch();
151         dialog.pack();
152         dialog.setLocationRelativeTo(dialog.getOwner());
153         dialog.setModal(true);
154         dialog.setVisible(true);
155         return lastReturnCode;
156     }
157 
158     /** Returns the last return code from the wizard dialog. */
159     public ReturnCode lastReturnCode() { return lastReturnCode; }
160 
161     @Override protected void onAfterStateSwitch() {
162         cancelButton.setText(format(wizard_cancel));
163 
164         backButton.setText(format(wizard_back));
165         backButton.setEnabled(switchBackEnabled());
166         requestFocusOnBackButtonIfEnabled();
167 
168         final WizardMessage nextMessage =
169                 switchNextEnabled() ? wizard_next : wizard_finish;
170         nextButton.setActionCommand(nextMessage.name());
171         nextButton.setText(format(nextMessage));
172         nextButton.setEnabled(true);
173         requestFocusOnNextButtonIfEnabled();
174 
175         cardLayout.show(cardPanel, currentState().name());
176     }
177 
178     private String format(WizardMessage key) { return key.format().toString(); }
179 
180     private void requestFocusOnBackButtonIfEnabled() {
181         if (backButton.isEnabled()) backButton.requestFocusInWindow();
182     }
183 
184     private void requestFocusOnNextButtonIfEnabled() {
185         if (nextButton.isEnabled()) nextButton.requestFocusInWindow();
186     }
187 
188     /**
189      * Closes the modal wizard dialog and sets the return code to
190      * {@link ReturnCode#cancel}.
191      */
192     protected void cancel() { close(ReturnCode.cancel); }
193 
194     /**
195      * Closes the modal wizard dialog and sets the return code to
196      * {@link ReturnCode#finish}.
197      */
198     protected void finish() { close(ReturnCode.finish); }
199 
200     private void close(final ReturnCode rc) {
201         dialog.setVisible(false);
202         lastReturnCode = rc;
203     }
204 }
