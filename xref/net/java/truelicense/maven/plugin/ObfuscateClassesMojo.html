6   package net.java.truelicense.maven.plugin;
7   
8   import de.schlichtherle.truezip.file.TFile;
9   import de.schlichtherle.truezip.file.TVFS;
10  import net.java.truelicense.maven.plugin.obfuscation.Processor;
11  import org.apache.maven.plugin.MojoFailureException;
12  import org.apache.maven.plugins.annotations.Parameter;
13  
14  import java.io.File;
15  import java.io.IOException;
16  import java.util.Formatter;
17  
18  import static java.io.File.createTempFile;
19  
20  /**
21   * A basic MOJO for the obfuscation of constant string values in Java class
22   * files (byte code).
23   *
24   * @see net.java.truelicense.obfuscate.ObfuscatedString
25   * @since TrueLicense 2.4
26   * @author Christian Schlichtherle
27   */
28  public abstract class ObfuscateClassesMojo extends MojoAdapter {
29  
30      @Parameter(defaultValue = "${project.build.directory}", readonly = true)
31      private File buildDirectory;
32  
33      /**
34       * Whether or not the class files shall get backed up to a JAR file before
35       * processing them.
36       *
37       * @deprecated This (non-)feature will be removed in TrueLicense 3.0.
38       */
39      @Deprecated
40      @Parameter(property = "truelicense.obfuscate.backup", defaultValue = "false")
41      private boolean backup;
42  
43      /**
44       * The prefix of the backup JAR file with the original class files.
45       *
46       * @deprecated This (non-)feature will be removed in TrueLicense 3.0.
47       */
48      @Deprecated
49      @Parameter(property = "truelicense.obfuscate.backupPrefix", defaultValue = "classes")
50      private String backupPrefix;
51  
52      /**
53       * The extension of the backup directory with the original class files:
54       * Use <code>"zip"</code> to create a ZIP file as the backup directory.
55       *
56       * @deprecated This (non-)feature will be removed in TrueLicense 3.0.
57       */
58      @Deprecated
59      @Parameter(property = "truelicense.obfuscate.backupExtension", defaultValue = "zip")
60      private String backupExtension;
61  
62      /** The maximum allowed size of a class file in bytes. */
63      @Parameter(property = "truelicense.obfuscate.maxBytes", defaultValue = "" + Processor.DEFAULT_MAX_BYTES)
64      private int maxBytes;
65  
66      /**
67       * The scope of constant string value obfuscation:
68       * <ul>
69       * <li>Use <code>none</code> to skip obfuscation.
70       * <li>Use <code>annotated</code> to obfuscate only constant string
71       *     values of fields which have been annotated with @Obfuscate.
72       * <li>Use <code>all</code> to obfuscate all constant string values.
73       *     This may result in quite some code size and runtime overhead,
74       *     so you should <em>not</em> use this in general.
75       * </ul>
76       */
77      @Parameter(property = "truelicense.obfuscate.scope", defaultValue = "annotated")
78      private Scope scope;
79  
80      /**
81       * The format for synthesized method names.
82       * This a format string for the class {@link Formatter}.
83       * It's first parameter is a string identifier for the obfuscation stage
84       * and its second parameter is an integer index for the synthesized method.
85       */
86      @Parameter(property = "truelicense.obfuscate.methodNameFormat", defaultValue = Processor.DEFAULT_METHOD_NAME_FORMAT)
87      private String methodNameFormat;
88  
89      /**
90       * Whether or not a call to <code>java.lang.String.intern()</code>
91       * shall get added when computing the original constant string values again.
92       * Use this to preserve the identity relation of constant string values if
93       * required.
94       */
95      @Parameter(property = "truelicense.obfuscate.intern", defaultValue = "" + Processor.DEFAULT_INTERN_STRINGS)
96      private boolean intern;
97  
98      protected abstract TFile outputDirectory();
99  
100     @Override
101     protected final void doExecute() throws MojoFailureException {
102         try {
103             if (scope == Scope.none) {
104                 getLog().warn("Skipping constant string value obfuscation.");
105             } else {
106                 getLog().info(String.format(
107                         "Obfuscating %s constant string values in %s.",
108                         scope,
109                         outputDirectory()));
110                 try {
111                     backup();
112                     obfuscate();
113                 } finally {
114                     TVFS.umount();
115                 }
116             }
117         } catch (Exception e) {
118             throw new MojoFailureException(e.toString(), e);
119         }
120     }
121 
122     @Deprecated
123     @SuppressWarnings("ResultOfMethodCallIgnored")
124     private void backup() throws IOException {
125         if (!backup) return;
126         final TFile backupDir = new TFile(createTempFile(
127                 backupPrefix + '.',
128                 '.' + backupExtension,
129                 buildDirectory));
130         getLog().info(backupDir + ": Creating class file backup directory.");
131         outputDirectory().cp_rp(backupDir.rm_r());
132     }
133 
134     private void obfuscate() {
135         Processor.builder()
136                 .logger(new LoggerAdapter(getLog()))
137                 .directory(outputDirectory())
138                 .maxBytes(maxBytes)
139                 .obfuscateAll(scope == Scope.all)
140                 .methodNameFormat(methodNameFormat)
141                 .internStrings(intern)
142                 .build()
143                 .run();
144     }
145 }
