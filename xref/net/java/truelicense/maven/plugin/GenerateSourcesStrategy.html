6   package net.java.truelicense.maven.plugin;
7   
8   import net.java.truelicense.maven.plugin.GenerateSourcesMojo.TemplateSet;
9   
10  /**
11   * Determines how the directory with the files generated by merging the
12   * template files with the Velocity context gets added to the maven project.
13   *
14   * @see GenerateSourcesMojo#generateSourcesStrategy()
15   * @author Christian Schlichtherle
16   */
17  public enum GenerateSourcesStrategy {
18  
19      mainSources {
20          void updateProjectFrom(TemplateSet templateSet) {
21              templateSet.addMergeDirectoryToCompileSourceRoot();
22          }
23      },
24  
25      testSources {
26          void updateProjectFrom(TemplateSet templateSet) {
27              templateSet.addMergeDirectoryToTestCompileSourceRoot();
28          }
29      };
30  
31      abstract void updateProjectFrom(TemplateSet templateSet);
32  }
