20  final class Obfuscator extends Visitor {
21  
22      private static final String
23              OBFUSCATED_STRING_INTERNAL_NAME = Type.getInternalName(ObfuscatedString.class);
24  
25      private final Map<String, ConstantStringReference>
26              csrs = new LinkedHashMap<String, ConstantStringReference>();
27  
28      /** Conversion method name. */
29      private final String cmn;
30  
31      Obfuscator(Processor ctx, ClassVisitor cv) {
32          super(ctx, cv);
33          cmn = super.internStrings() ? "toStringIntern" : "toString";
34      }
35  
36      @Override public @CheckForNull FieldVisitor visitField(
37              int access,
38              String name,
39              String desc,
40              @CheckForNull String signature,
41              @CheckForNull Object value) {
42          return new O9n2ndFieldNode(access, name, desc, signature, value);
43      }
44  
45      @Override public @CheckForNull MethodVisitor visitMethod(
46              final int access,
47              final String name,
48              final String desc,
49              final @CheckForNull String signature,
50              final @CheckForNull String[] exceptions) {
51          final MethodVisitor mv = cv.visitMethod(access, name, desc, signature, exceptions);
52          return null == mv
53                  ? null
54                  : "<init>".equals(name)
55                      ? new O9n2ndConstructorVisitor(name, mv)
56                      : new O9n2ndMethodVisitor(name, mv);
57      }
58  
59      @Override public void visitEnd() {
60          for (final Entry<String, ConstantStringReference> e : csrs.entrySet())
61              addComputationMethod(e.getValue());
62          cv.visitEnd();
63      }
64  
65      private void addComputationMethod(final ConstantStringReference csr) {
66          final int access = csr.access;
67          final boolean keepField = csr.keepField();
68          final boolean isStatic = csr.isStatic();
69          final String name = csr.name;
70          final String value = csr.value;
71          final String lvalue = literal(value);
72          final MethodVisitor mv;
73          if (keepField && isStatic) {
74              mv = cv.visitMethod(access, "<clinit>", "()V", null, null);
75              if (null == mv) return;
76              logger().debug("Adding method <clinit> to compute \"{}\" and put it into field {}.", lvalue, name);
77          } else {
78              mv = cv.visitMethod(access, name, "()Ljava/lang/String;", null, null);
79              if (null == mv) return;
80              logger().debug("Adding method {} to compute \"{}\".", name, lvalue);
81          }
82          mv.visitCode();
83          mv.visitTypeInsn(NEW, OBFUSCATED_STRING_INTERNAL_NAME);
84          mv.visitInsn(DUP);
85          final long[] obfuscated = array(value);
86          addInsn(mv, obfuscated.length);
87          mv.visitIntInsn(NEWARRAY, T_LONG);
88          for (int i = 0; i < obfuscated.length; i++) {
89              mv.visitInsn(DUP);
90              addInsn(mv, i);
91              mv.visitLdcInsn(obfuscated[i]);
92              mv.visitInsn(LASTORE);
93          }
94          mv.visitMethodInsn(INVOKESPECIAL, OBFUSCATED_STRING_INTERNAL_NAME, "<init>", "([J)V", false);
95          mv.visitMethodInsn(INVOKEVIRTUAL, OBFUSCATED_STRING_INTERNAL_NAME, cmn, "()Ljava/lang/String;", false);
96          if (keepField && isStatic) {
97              mv.visitFieldInsn(PUTSTATIC, internalClassName(), name, "Ljava/lang/String;");
98              mv.visitInsn(RETURN);
99              mv.visitMaxs(7, 0);
100         } else {
101             mv.visitInsn(ARETURN);
102             mv.visitMaxs(7, isStatic ? 0 : 1);
103         }
104         mv.visitEnd();
105     }
106 
107     private static void addInsn(final MethodVisitor mv, final int index) {
108         switch (index) {
109             case 0:  mv.visitInsn(ICONST_0); break;
110             case 1:  mv.visitInsn(ICONST_1); break;
111             case 2:  mv.visitInsn(ICONST_2); break;
112             case 3:  mv.visitInsn(ICONST_3); break;
113             case 4:  mv.visitInsn(ICONST_4); break;
114             case 5:  mv.visitInsn(ICONST_5); break;
115             default: mv.visitIntInsn(BIPUSH, index);
116         }
117     }

119     final ConstantStringReference register(
120             final ConstantStringReferenceFactory factory,
121             final String value) {
122         final Map<String, ConstantStringReference> csrs = this.csrs;
123         ConstantStringReference csr = csrs.get(value);
124         if (null != csr) return csr;
125         csrs.put(value, csr = factory.csr(value));
126         return csr;
127     }
128 
129     @Immutable
130     private static final class ConstantStringReference {
131         final int access;
132         final String name, value;
133 
134         ConstantStringReference(final int access, final String name, final String value) {
135             this.access = access | ACC_SYNTHETIC;
136             this.name = name;
137             this.value = value;
138         }
139 
140         boolean keepField() {
141             return isProtectedOrPublic() || !isStatic();
142         }
143 
144         boolean isPrivate() { return 0 != (access & ACC_PRIVATE); }
145 
146         boolean isProtectedOrPublic() {
147             return 0 != (access & (ACC_PROTECTED | ACC_PUBLIC));
148         }
149 
150         boolean isStatic() { return 0 != (access & ACC_STATIC); }
151     }
152 
153     private interface ConstantStringReferenceFactory {
154         ConstantStringReference csr(String csv);
155     }
156 
157     private final class O9n2ndFieldNode
158     extends O9nFieldNode implements ConstantStringReferenceFactory {
159 
160         O9n2ndFieldNode(
161                 int access,
162                 String name,
163                 String desc,
164                 @CheckForNull String signature,
165                 @CheckForNull Object value) {
166             super(access, name, desc, signature, value);
167         }
168 
169         @Override public void visitEnd() {
170             final boolean no = needsObfuscation;
171             if (obfuscateAll() || no) {
172                 final String value = stringValue;
173                 if (null != value) {
174                     final ConstantStringReference csr = register(this, value);
175                     final String name = csr.name; // shadow field name
176                     if (csr.keepField()) {
177                         if (no)
178                             logger().warn("Obfuscation of protected or public or non-static field {} is insecure because it can't get removed from the byte code.", name);
179                         // Fall through.
180                     } else {
181                         logger().debug("Removing private or package-private static field {}.", name);
182                         return; // remove field
183                     }
184                 } else if (no) {
185                     logger().error("Annotated field {} does not have a constant string value.", name);
186                     // Fall through.
187                 }
188             }
189             // Keep field for subsequent processing.
190             accept(cv);
191         }
192 
193         @Override public ConstantStringReference csr(String value) {
194             return new ConstantStringReference(access, name, value);
195         }
196     }

198     private class O9n2ndMethodVisitor
199     extends MethodVisitor implements ConstantStringReferenceFactory {
200 
201         final String localMethodName;
202 
203         O9n2ndMethodVisitor(final String name, MethodVisitor mv) {
204             super(ASM5, mv);
205             this.localMethodName = name;
206         }
207 
208         @Override public void visitLdcInsn(final Object cst) {
209             final String value;
210             if (cst instanceof String && replace(value = (String) cst)) {
211                 process(register(this, value));
212             } else {
213                 mv.visitLdcInsn(cst);
214             }
215         }
216 
217         boolean replace(String value) {
218             return obfuscateAll() || constantStrings().contains(value);
219         }
220 
221         void process(ConstantStringReference csr) {
222             final boolean isStatic = csr.isStatic();
223             final String name = csr.name;
224             final String lvalue = literal(csr.value);
225             final String lmn = this.localMethodName;
226             final String icn = internalClassName();
227             if (!isStatic) mv.visitVarInsn(ALOAD, 0);
228             if (readField(csr)) {
229                 logger().debug("Replacing \"{}\" in method {} with access to field {}.", lvalue, lmn, name);
230                 final int opcode = isStatic ? GETSTATIC : GETFIELD;
231                 mv.visitFieldInsn(opcode, icn, name, "Ljava/lang/String;");
232             } else {
233                 logger().debug("Replacing \"{}\" in method {} with call to method {}.", lvalue, lmn, name);
234                 final int opcode = isStatic
235                         ? INVOKESTATIC : csr.isPrivate()
236                             ? INVOKESPECIAL : INVOKEVIRTUAL;
237                 mv.visitMethodInsn(opcode, icn, name, "()Ljava/lang/String;", false);
238             }
239         }
240 
241         boolean readField(ConstantStringReference csr) {
242             return csr.keepField();
243         }
244 
245         @Override public ConstantStringReference csr(String value) {
246             return new ConstantStringReference(
247                     ACC_PRIVATE | ACC_STATIC,
248                     methodName("string", csrs.size()),
249                     value);
250         }
251     }
252 
253     private class O9n2ndConstructorVisitor extends O9n2ndMethodVisitor {
254 
255         O9n2ndConstructorVisitor(String name, MethodVisitor mv) {
256             super(name, mv);
257         }
258 
259         @Override boolean readField(ConstantStringReference csr) {
260             return csr.isProtectedOrPublic() && csr.isStatic();
261         }
262     }
