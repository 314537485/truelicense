6   package net.java.truelicense.maven.plugin.obfuscation;
7   
8   import de.schlichtherle.truezip.file.TFileInputStream;
9   import de.schlichtherle.truezip.file.TFileOutputStream;
10  import java.io.*;
11  import javax.annotation.concurrent.NotThreadSafe;
12  
13  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
14  import org.slf4j.Logger;
15  
16  /**
17   * @author Christian Schlichtherle
18   */
19  @NotThreadSafe
20  abstract class Pass implements Runnable {
21  
22      final Processor ctx;
23  
24      Pass(final Processor ctx) {
25          this.ctx = ctx;
26      }
27  
28      final Logger logger() { return ctx.logger(); }
29  
30      final Logger logger(Node node) { return ctx.logger(node.path()); }
31  
32      final boolean obfuscateAll() { return ctx.obfuscateAll(); }
33  
34      @Override
35      public void run() { scan(new Node("", ctx.directory())); }
36  
37      private void scan(final Node node) {
38          final File file = node.file();
39          if (file.isDirectory()) {
40              final String[] members = file.list();
41              if (null != members) {
42                  for (String member : members)
43                      scan(new Node(node, member));
44              } else {
45                  logger(node).error("Cannot list directory.");
46              }
47          } else if (file.isFile() && file.getName().endsWith(".class")) {
48              process(node);
49          } else {
50              logger(node).trace("Skipping resource file.");
51          }
52      }
53  
54      abstract void process(final Node node);
55  
56      @SuppressFBWarnings("OS_OPEN_STREAM")
57      final byte[] read(final Node node) throws IOException {
58          final File file = node.file();
59          final byte[] code;
60          {
61              final long l = file.length();
62              final int mb = ctx.maxBytes();
63              if (l > mb)
64                  throw new IOException(
65                          String.format("%s (%,d bytes exceeds max %,d bytes)", file, l, mb));
66              code = new byte[(int) l];
67          }
68          {
69              final InputStream in = new TFileInputStream(file);
70              try { new DataInputStream(in).readFully(code); }
71              finally { in.close(); }
72          }
73          return code;
74      }
75  
76      @SuppressFBWarnings("OBL_UNSATISFIED_OBLIGATION")
77      final void write(final Node node, final byte[] code) throws IOException {
78          final OutputStream out = new TFileOutputStream(node.file());
79          try { out.write(code); }
80          finally { out.close(); }
81      }
82  }
