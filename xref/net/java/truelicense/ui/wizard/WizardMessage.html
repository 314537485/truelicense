10  /**
11   * Enumerates the standard messages of any wizard dialog.
12   *
13   * @author Christian Schlichtherle
14   * @since  TrueLicense 2.3
15   */
16  public enum WizardMessage implements Formattable {
17  
18      wizard_back, wizard_next, wizard_cancel, wizard_finish;
19  
20      @Override public Message format(Object... args) {
21          return Messages.message(name(), args);
22      }
23  }
