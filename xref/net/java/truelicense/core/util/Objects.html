6   package net.java.truelicense.core.util;
7   
8   import javax.annotation.CheckForNull;
9   import javax.annotation.concurrent.Immutable;
10  
11  /**
12   * Duplicates some features of {@code java.util.Objects} in JSE 7.
13   * May be removed when migrating to JSE 7 one day.
14   *
15   * @author Christian Schlichtherle
16   */
17  @Immutable
18  public class Objects {
19  
20      public static boolean equals(@CheckForNull Object a, @CheckForNull Object b) {
21          return a == b || null != a && a.equals(b);
22      }
23  
24      public static int hashCode(@CheckForNull Object o) {
25          return null == o ? 0 : o.hashCode();
26      }
27  
28      public static <T> T requireNonNull(final @CheckForNull T obj) {
29          if (null == obj) throw new NullPointerException();
30          return obj;
31      }
32  
33      private Objects() { }
34  }
