6   package net.java.truelicense.core;
7   
8   import net.java.truelicense.core.io.Source;
9   import net.java.truelicense.core.util.ContextProvider;
10  
11  /**
12   * Defines the life cycle management operations for license keys in consumer
13   * applications.
14   *
15   * <h3>How to Preview License Keys</h3>
16   * <p>
17   * This interface intentionally lacks a method to preview the license bean which is encoded in a license key.
18   * You can work around this constraint by using a {@linkplain LicenseConsumerContext license consumer context} to
19   * configure a license consumer manager which uses a transient {@link net.java.truelicense.core.io.MemoryStore} instead
20   * of a persistent store - see {@link LicenseConsumerContext.ManagerBuilder#storeIn}.
21   * Once configured, you can {@linkplain #install install} the license key to the transient memory store and
22   * {@linkplain #view view} its encoded license bean.
23   *
24   * @author Christian Schlichtherle
25   */
26  public interface LicenseConsumerManager
27  extends ContextProvider<LicenseConsumerContext>,
28          LicenseParametersProvider,
29          LicenseSubjectProvider {
30  
31      /**
32       * Installs the license key from the given source and returns an
33       * unvalidated duplicate of its encoded license bean.
34       * Unlike {@link #verify}, this operation does not validate the license
35       * bean.
36       * This enables the caller to obtain a duplicate of the license bean even
37       * if its validation would fail, e.g. if the license has expired.
38       * Note that this is a change of the behavior from versions prior to
39       * TrueLicense 2.4 where license validation was mandatory.
40       * <p>
41       * Calling this operation performs an initial
42       * {@linkplain LicenseAuthorization#clearInstall authorization check}.
43       *
44       * @param  source the source for loading the license key.
45       * @return An unvalidated duplicate of the license bean which is encoded in
46       *         the license key.
47       */
48      License install(Source source) throws LicenseManagementException;
49  
50      /**
51       * Returns an unvalidated duplicate of the license bean which is encoded in
52       * the installed license key.
53       * You should call this method whenever you want to display the properties
54       * of the installed license.
55       * Unlike {@link #verify}, this operation does <em>not</em> validate the
56       * license bean.
57       * This enables the caller to obtain a duplicate of the license bean even
58       * if its validation would fail, e.g. if the license has expired.
59       * <p>
60       * Calling this operation performs an initial
61       * {@linkplain LicenseAuthorization#clearView authorization check}.
62       *
63       * @return An unvalidated duplicate of the license bean which is encoded in
64       *         the installed license key.
65       */
66      License view() throws LicenseManagementException;
67  
68      /**
69       * Verifies the license bean which is encoded in the installed license key.
70       * You should call this method whenever you want to verify access to a
71       * feature of your software product.
72       * Execution needs to be fast in order to support frequent calling.
73       * <p>
74       * Calling this operation performs an initial
75       * {@linkplain LicenseAuthorization#clearVerify authorization check}.
76       *
77       * @throws LicenseValidationException if validating the license bean fails,
78       *         e.g. if the license has expired.
79       */
80      void verify() throws LicenseManagementException;
81  
82      /**
83       * Uninstalls the installed license key.
84       * <p>
85       * Calling this operation performs an initial
86       * {@linkplain LicenseAuthorization#clearUninstall authorization check}.
87       */
88      void uninstall() throws LicenseManagementException;
89  }
