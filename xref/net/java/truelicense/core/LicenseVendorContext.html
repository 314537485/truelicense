6   package net.java.truelicense.core;
7   
8   import net.java.truelicense.core.auth.Authentication;
9   import net.java.truelicense.core.codec.CodecProvider;
10  import net.java.truelicense.core.crypto.Encryption;
11  import net.java.truelicense.core.io.Source;
12  import net.java.truelicense.core.util.Builder;
13  import net.java.truelicense.core.util.Injection;
14  import net.java.truelicense.obfuscate.Obfuscate;
15  import net.java.truelicense.obfuscate.ObfuscatedString;
16  
17  import javax.annotation.CheckForNull;
18  
19  /**
20   * A derived context for license vendor applications alias license key tools.
21   * Use this context to configure a {@link LicenseVendorManager} with the
22   * required parameters.
23   * For a demonstration of this API, please use the TrueLicense Maven Archetype
24   * to generate a sample project - even if you don't use Maven to build your
25   * software product.
26   * <p>
27   * Applications have no need to implement this interface and should not do so
28   * because it may be subject to expansion in future versions.
29   *
30   * @author Christian Schlichtherle
31   */
32  public interface LicenseVendorContext
33  extends CodecProvider, LicenseApplicationContext, LicenseProvider {
34  
35      /**
36       * Configures a license vendor manager.
37       *
38       * @param  authentication the authentication.
39       * @param  encryption the encryption.
40       * @return A license vendor manager.
41       */
42      LicenseVendorManager manager(Authentication authentication, Encryption encryption);
43  
44      /**
45       * Configures a key store based authentication.
46       * <p>
47       * The provided strings should be computed on demand from an obfuscated
48       * form, e.g. by annotating a constant string value with the
49       * &#64;{@link Obfuscate} annotation and processing it with the TrueLicense
50       * Maven Plugin.
51       *
52       * @param source the nullable source for the key store.
53       *               May be {@code null} if and only if the key store type
54       *               does not require loading from an input stream.
55       * @param storeType the nullable type of the key store,
56       *                  e.g. {@code "JCEKS"} or {@code "JKS"}.
57       *                  If this is {@code null}, then the value computed by the
58       *                  expression
59       *                  <code>{@link #context()}.{@link LicenseManagementContext#storeType() storeType()}</code>
60       *                  is used instead.
61       *                  Otherwise, you need to make sure that the store type is
62       *                  supported by a security provider on all platforms - see
63       *                  <a href="http://docs.oracle.com/javase/6/docs/technotes/guides/security/SunProviders.html">Java Cryptography Architeture Sun Providers Documentation</a>.
64       * @param storePassword the password for verifying the integrity of the
65       *                      key store.
66       * @param alias the alias of the private key entry in the key store.
67       * @param keyPassword the nullable password for accessing the private key
68       *                    in the key entry.
69       *                    If this is {@code null} or empty, then the
70       *                    {@code storePassword} is used instead.
71       * @return A key store based authentication.
72       * @throws IllegalArgumentException if a password is considered to be too
73       *         weak.
74       */
75      Authentication keyStore(@CheckForNull Source source, @CheckForNull String storeType, ObfuscatedString storePassword, String alias, @CheckForNull ObfuscatedString keyPassword);
76  
77      /**
78       * {@inheritDoc}
79       *
80       * @throws IllegalArgumentException if the password is considered to be too
81       *         weak.
82       */
83      @Override Encryption pbe(String algorithm, ObfuscatedString password);
84  
85      /**
86       * Returns a builder for a
87       * {@linkplain LicenseVendorManager license vendor manager}.
88       * Call its {@link ManagerBuilder#build} method to obtain a configured
89       * license vendor manager.
90       */
91      ManagerBuilder manager();
92  
93      /**
94       * A builder for
95       * {@linkplain LicenseVendorManager license vendor managers}.
96       * Call its {@link #build} method to obtain a configured license vendor
97       * manager.
98       *
99       * @author Christian Schlichtherle
100      */
101     interface ManagerBuilder extends Builder<LicenseVendorManager> {
102 
103         /**
104          * Sets the authentication.
105          *
106          * @return {@code this}.
107          */
108         ManagerBuilder authentication(Authentication authentication);
109 
110         /**
111          * Returns an injection for a key store based authentication.
112          * Call its {@link Injection#inject} method to build and inject the
113          * configured authentication into this builder and return it.
114          * <p>
115          * The keystore needs to have a key password configured for the private
116          * key entry.
117          *
118          * @see #authentication(Authentication)
119          */
120         KsbaInjection<ManagerBuilder> keyStore();
121 
122         /**
123          * Sets the encryption.
124          *
125          * @return {@code this}.
126          */
127         ManagerBuilder encryption(Encryption encryption);
128 
129         /**
130          * Returns an injection for a password based encryption.
131          * Call its {@link Injection#inject} method to build and inject the
132          * configured encryption into this builder and return it.
133          *
134          * @since TrueLicense 2.5
135          * @see #encryption(Encryption)
136          */
137         PbeInjection<ManagerBuilder> encryption();
138 
139         /**
140          * @deprecated since TrueLicense 2.5. Use {@link #encryption()} instead.
141          */
142         @Deprecated
143         PbeInjection<ManagerBuilder> pbe();
144     }
145 }
