6   package net.java.truelicense.core.io;
7   
8   /**
9    * A transformation decorates the {@link java.io.OutputStream}s and
10   * {@link java.io.InputStream}s which are provided by {@linkplain Sink sinks}
11   * and {@linkplain Source sources}.
12   * <p>
13   * For example, in an encryption transformation, the method {@link #apply}
14   * could decorate the output streams provided by the given sink with a new
15   * {@link javax.crypto.CipherOutputStream} in order to encrypt the data before
16   * writing it to the underlying output stream.
17   * Likewise, the method {@link #unapply} could decorate the input streams
18   * provided by the given source with a new
19   * {@link javax.crypto.CipherInputStream} in order to decrypt the data after
20   * reading it from the underlying input stream.
21   * <p>
22   * As another example, in a compression transformation, the apply method could
23   * decorate the output streams provided by the given sink with a new
24   * {@link java.util.zip.DeflaterOutputStream} in order to compress the data
25   * before writing it to the underlying output stream.
26   * Likewise, the unapply method could decorate the input streams provided by the
27   * given source with a new {@link java.util.zip.InflaterInputStream} in order to
28   * decompress the data after reading it from the underlying input stream.
29   * <p>
30   * The benefit of this interface is that you can easily chain the apply and
31   * unapply methods in order to create rich decorators for the underlying sinks
32   * and sources without needing to know anything about the implementation of the
33   * transformations.
34   * <p>
35   * For example, depending on the previous examples, the following test code
36   * would assert the round-trip processing of the string {@code "Hello world!"}
37   * using the composition of some compression and encryption transformations on
38   * some store:
39   * <pre>{@code
40   * Transformation compression = ...;
41   * Transformation encryption = ...;
42   * Store store = ...;
43   * Sink compressAndEncryptData = compression.apply(encryption.apply(store));
44   * Source decryptAndDecompressData = compression.unapply(encryption.unapply(store));
45   * try (PrintWriter writer = new PrintWriter(compressAndEncryptData.output())) {
46   *     writer.println("Hello world!");
47   * }
48   * try (BufferedReader reader = new BufferedReader(new InputStreamReader(decryptAndDecompressData.input()))) {
49   *     assertTrue("Hello world!".equals(reader.readLine()));
50   * }
51   * }</pre>
52   *
53   * @author Christian Schlichtherle
54   */
55  public interface Transformation {
56  
57      /**
58       * Returns a sink which decorates the output streams provided by the given
59       * sink.
60       *
61       * @param  sink the sink which provides the output streams to decorate.
62       */
63      Sink apply(Sink sink);
64  
65      /**
66       * Returns a source which decorates the input streams provided by the given
67       * source.
68       *
69       * @param  source the source which provides the input streams to decorate.
70       */
71      Source unapply(Source source);
72  }
