6   package net.java.truelicense.core.codec;
7   
8   import java.lang.reflect.Type;
9   import javax.annotation.Nullable;
10  import net.java.truelicense.core.io.*;
11  
12  /**
13   * Defines an object graph encoding/decoding (alias serialization).
14   *
15   * @author Christian Schlichtherle
16   */
17  public interface Codec {
18  
19      /**
20       * Returns an identifier for the content type used by this codec.
21       * The returned string must conform to the syntax specified in
22       * <a href="http://tools.ietf.org/html/rfc2045">RFC2045</a>
23       * "Multipurpose Internet Mail Extensions (MIME) Part One: Format of Internet Message Bodies",
24       * <a href="http://tools.ietf.org/html/rfc2045#section-5.1">Section 5.1</a>
25       * "Syntax of the Content-Type Header Field",
26       * except that it must not start with "Content-Type:"
27       * and optional spaces, i.e. the field name must be stripped.
28       */
29      String contentType();
30  
31      /**
32       * Returns an identifier for the content transfer encoding used by this
33       * codec.
34       * The returned string must conform to the syntax specified in
35       * <a href="http://tools.ietf.org/html/rfc2045">RFC2045</a>
36       * "Multipurpose Internet Mail Extensions (MIME) Part One: Format of Internet Message Bodies",
37       * <a href="http://tools.ietf.org/html/rfc2045#section-6.1">Section 6.1</a>
38       * "Content-Transfer-Encoding Syntax",
39       * except that it must not start with "Content-Transfer-Encoding:"
40       * and optional spaces, i.e. the field name must be stripped.
41       * <p>
42       * If the Content-Transfer-Encoding equals (ignoring case) {@code "8bit"}
43       * and the  {@linkplain #contentType Content-Type} does not specify a
44       * {@code charset} parameter, then {@code UTF-8} is assumed as the charset.
45       */
46      String contentTransferEncoding();
47  
48      /**
49       * Encodes a nullable object graph to the given sink.
50       *
51       * @param sink the sink to write the encoded object graph to.
52       * @param obj the nullable object graph.
53       *        Implementations should support encoding {@code null}.
54       *        If they do not support this, then this should be documented in
55       *        the Javadoc.
56       */
57      void encode(Sink sink, @Nullable Object obj) throws Exception;
58  
59      /**
60       * Decodes a nullable object graph from the given source.
61       *
62       * @param  <T> the expected generic type of the decoded object.
63       * @param  source the source from where to read the encoded object graph
64       *         from.
65       * @param  expected the expected generic type of the decoded object graph,
66       *         e.g. {@code String.class}.
67       *         This is just a hint and the implementation may ignore it.
68       * @return A duplicate of the original object graph.
69       *         Its actual type may differ from the expected generic type.
70       *         It may be {@code null} if and only if the original object graph
71       *         was {@code null}.
72       */
73      @Nullable <T> T decode(Source source, Type expected) throws Exception;
74  }
