8   import java.nio.charset.*;
9   import java.util.regex.Matcher;
10  import java.util.regex.Pattern;
11  import javax.annotation.CheckForNull;
12  
13  import net.java.truelicense.core.io.MemoryStore;
14  import net.java.truelicense.core.io.Store;
15  import net.java.truelicense.obfuscate.Obfuscate;
16  import org.apache.commons.codec.Charsets;
17  
18  /**
19   * Provides common {@link Codec} functions.
20   *
21   * @author Christian Schlichtherle
22   */
23  public class Codecs {
24  
25      @Obfuscate static final String APPLICATION_XML_WITH_UTF_8 = "application/xml; charset=utf-8";
26  
27      @Obfuscate static final String SEVEN_BIT = "7bit";
28      @Obfuscate static final String EIGHT_BIT = "8bit";
29      @Obfuscate static final String QUOTED_PRINTABLE = "quoted-printable";
30      @Obfuscate static final String BASE64 = "base64";
31  
32      /** Not perfect, but should work for all valid input. */
33      @Obfuscate private static final String CHARSET_REGEXP =
34              "\\s*charset\\s*=\\s*(?:\"([^\"]*)\"|([^\\s\\(\\)<>@,;:\\\\\"/\\[\\]\\?=]+))";
35  
36      private static final Pattern CHARSET_PATTERN =
37              Pattern.compile(CHARSET_REGEXP, Pattern.CASE_INSENSITIVE);
38  
39      /**
40       * Figures the content transfer charset which is used by the given codec.
41       * This method passes the call to {@link #contentTransferCharset}.
42       * If an {@link IllegalArgumentException} is thrown, then it returns
43       * {@code null}.
44       *
45       * @param  codec the codec to test.
46       * @return The content transfer charset which is used by the given codec or
47       *         {@code null} if the codec doesn't produce text or specifies an
48       *         invalid charset name or an unknown charset.
49       */
50      public static @CheckForNull Charset charset(final Codec codec) {
51          try { return contentTransferCharset(codec); }
52          catch (IllegalArgumentException ex) { return null; }
53      }
54  
55      /**
56       * Figures the content transfer charset which is used by the given codec.
57       * <ol>
58       * <li>
59       * First, the codec's
60       * {@linkplain Codec#contentTransferEncoding Content-Transfer-Encoding}
61       * gets tested:
62       * If it is {@code "7bit"}, {@code "quoted-printable"} or {@code "base64"}
63       * (ignoring case), then the returned charset is US-ASCII.
64       * <li>
65       * Otherwise, if the codec's Content-Transfer-Encoding is {@code "8bit"}
66       * (ignoring case), then the codec's {@link Codec#contentType Content-Type}
67       * gets tested:
68       * <ol>
69       * <li>
70       * If the codec's Content-Type specifies a {@code charset} parameter
71       * (ignoring case), then an attempt is made to load this charset.
72       * Upon success, this charset gets returned.
73       * <li>
74       * If the codec's Content-Type does not specify a {@code charset} parameter,
75       * then UTF-8 gets returned.
76       * This ensures compatibility with JSON.
77       * </ol>
78       * <li>
79       * Finally, in all other cases, an {@link IllegalArgumentException} gets
80       * thrown in order to specify that the codec doesn't produce text or
81       * specifies an invalid charset name or an unknown charset.
82       * </ol>
83       *
84       * @param  codec the codec to test.
85       * @return The content transfer charset which is used by the given codec.
86       * @throws IllegalCharsetNameException if the specified charset name is
87       *         invalid.
88       * @throws UnsupportedCharsetException if the specified charset is unknown.
89       * @throws BinaryCodecException if the codec doesn't produce text.
90       * @since TrueLicense 2.2.1
91       */
92      public static Charset contentTransferCharset(final Codec codec) {
93          final String encoding = codec.contentTransferEncoding();
94          if (    SEVEN_BIT.equalsIgnoreCase(encoding) ||
95                  QUOTED_PRINTABLE.equalsIgnoreCase(encoding) ||
96                  BASE64.equalsIgnoreCase(encoding)) {
97              return Charsets.US_ASCII;
98          } else  if (EIGHT_BIT.equalsIgnoreCase(encoding)) {
99              final Matcher matcher = CHARSET_PATTERN.matcher(codec.contentType());
100             if (matcher.find()) {
101                 assert 2 == matcher.groupCount();
102                 String charset = matcher.group(1);
103                 if (null == charset) charset = matcher.group(2);
104                 return Charset.forName(charset);
105             } else {
106                 return Charsets.UTF_8;
107             }
108         }
109         throw new BinaryCodecException(
110                 "The Content-Transfer-Encoding " + encoding + " doesn't produce text.");
111     }
112 
113     /**
114      * Returns a clone of the given object using a new
115      * {@link SerializationCodec}.
116      *
117      * @since TrueLicense 2.5
118      */
119     public static <T> T clone(T object) throws Exception {
120         return clone(object, new SerializationCodec());
121     }
122 
123     /**
124      * Returns a clone of the given object using the given codec.
125      *
126      * @since TrueLicense 2.5
127      */
128     public static <T> T clone(final T object, final Codec codec) throws Exception {
129         final Store store = new MemoryStore();
130         codec.encode(store, object);
131         return codec.decode(store, object.getClass());
132     }
133 
134     private Codecs() { }
135 }
