6   package net.java.truelicense.core.auth;
7   
8   import net.java.truelicense.core.codec.Codec;
9   
10  import javax.annotation.Nullable;
11  import javax.annotation.concurrent.Immutable;
12  import java.security.PrivateKey;
13  import java.security.PublicKey;
14  import java.security.Signature;
15  
16  /**
17   * A controller for storing authenticated objects (artifacts).
18   * This interface is used by {@linkplain Authentication authentications} to
19   * {@linkplain Authentication#sign sign} and
20   * {@linkplain Authentication#verify verify} artifacts.
21   *
22   * @author Christian Schlichtherle
23   */
24  @Immutable
25  public interface Repository {
26  
27      /** Returns the repository model. */
28      RepositoryModel model();
29  
30      /**
31       * Encodes and signs the given {@code artifact}
32       * and returns an artifactory for decoding it.
33       * As a side effect, the state of this repository is modified so that a
34       * subsequent {@linkplain #verify verification} can succeed.
35       *
36       * @param  codec the codec for encoding the artifact.
37       *         It is an implementation detail whether or not encoding
38       *         {@code null} is supported or not.
39       * @param  engine the signature engine.
40       * @param  key the private key.
41       * @param  artifact the nullable artifact to sign.
42       *         This may be {@code null} if and only if the {@code codec}
43       *         supports encoding it.
44       */
45      Artifactory sign(
46              Codec codec,
47              Signature engine,
48              PrivateKey key,
49              @Nullable Object artifact)
50      throws Exception;
51  
52      /**
53       * Verifies the signature of the encoded artifact
54       * and returns an artifactory for decoding it.
55       * The state of this repository is not modified by this method.
56       *
57       * @param  codec the codec for decoding the artifact.
58       *         It is an implementation detail whether or not decoding
59       *         {@code null} is supported or not.
60       * @param  engine the signature engine.
61       * @param  key the public key.
62       * @throws RepositoryIntegrityException if the integrity of the repository
63       *         with its encoded artifact has been compromised.
64       */
65      Artifactory verify(
66              Codec codec,
67              Signature engine,
68              PublicKey key)
69      throws Exception;
70  }
