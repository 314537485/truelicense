6   package net.java.truelicense.core.auth;
7   
8   import net.java.truelicense.core.io.Source;
9   import net.java.truelicense.core.io.SourceProvider;
10  import net.java.truelicense.obfuscate.Obfuscate;
11  
12  import javax.annotation.CheckForNull;
13  import java.util.Arrays;
14  
15  /**
16   * Defines parameters for accessing a {@link java.security.KeyStore} which
17   * holds the public or private keys to sign or verify a {@link Repository}
18   * by an {@link Authentication}.
19   * <p>
20   * Unless stated otherwise, all no-argument methods need to return consistent
21   * objects so that caching them is not required.
22   * A returned object is considered to be consistent if it compares
23   * {@linkplain Object#equals(Object) equal} or at least behaves identical to
24   * any previously returned object.
25   *
26   * @author Christian Schlichtherle
27   */
28  public interface KeyStoreParameters
29  extends AuthenticationParameters, SourceProvider {
30  
31      /** Returns the alias of the entry in the key store. */
32      String alias();
33  
34      /**
35       * Returns a new char array with the password for accessing the private key
36       * in the key entry.
37       * If the returned array is empty, then the
38       * {@linkplain #storePassword() key store password} should get used instead.
39       * <p>
40       * It is the caller's responsibility to wipe the contents of the char array
41       * after use, e.g. by a call to {@link Arrays#fill(char[], char)}.
42       *
43       * @return A new char array with the password for accessing the private key
44       *         in the key entry.
45       */
46      char[] keyPassword();
47  
48      /**
49       * Returns the nullable input source for the key store.
50       *
51       * @return the nullable input source for the key store.
52       *         May be {@code null} if and only if the key store type does not
53       *         require loading from an input stream.
54       */
55      @Override
56      @CheckForNull Source source();
57  
58      /**
59       * Returns a new char array with the password for verifying the integrity
60       * of the key store.
61       * <p>
62       * It is the caller's responsibility to wipe the contents of the char array
63       * after use, e.g. by a call to {@link Arrays#fill(char[], char)}.
64       *
65       * @return A new char array with the password for verifying the key store.
66       */
67      char[] storePassword();
68  
69      /**
70       * Returns the type of the key store,
71       * for example {@code "JCEKS"} or {@code "JKS"}.
72       * The returned string should be computed on demand from an obfuscated form,
73       * e.g. by annotating a constant string value with the \@{@link Obfuscate}
74       * annotation and processing it with the TrueLicense Maven Plugin.
75       */
76      String storeType();
77  }
