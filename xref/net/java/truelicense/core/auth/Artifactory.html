6   package net.java.truelicense.core.auth;
7   
8   import javax.annotation.Nullable;
9   import java.lang.reflect.Type;
10  
11  /**
12   * An immutable factory for authenticated objects (artifacts).
13   *
14   * @author Christian Schlichtherle
15   */
16  public interface Artifactory {
17  
18      /**
19       * Decodes the artifact from its encoded form.
20       *
21       * @param  <T> the expected generic type of the decoded artifact.
22       * @param  expected the expected generic type of the decoded artifact,
23       *         e.g. {@code String.class}.
24       *         This is just a hint and the implementation may ignore it.
25       * @return A duplicate of the original artifact.
26       *         Its actual type may differ from the expected generic type.
27       *         This may be {@code null} if and only if the original artifact
28       *         was {@code null}.
29       */
30      @Nullable <T> T decode(Type expected) throws Exception;
31  }
