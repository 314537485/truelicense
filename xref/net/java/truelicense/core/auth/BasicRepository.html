6   package net.java.truelicense.core.auth;

8   import net.java.truelicense.core.codec.Codec;
9   import net.java.truelicense.core.io.MemoryStore;
10  import org.apache.commons.codec.binary.Base64;
11  
12  import javax.annotation.CheckForNull;
13  import javax.annotation.Nullable;
14  import javax.xml.bind.annotation.XmlRootElement;
15  import javax.xml.bind.annotation.XmlType;
16  import java.nio.charset.Charset;
17  import java.security.PrivateKey;
18  import java.security.PublicKey;
19  import java.security.Signature;
20  
21  import static net.java.truelicense.core.codec.Codecs.charset;
22  
23  /**
24   * A basic repository for storing authenticated objects (artifacts).
25   *
26   * @author Christian Schlichtherle
27   */
28  // #TRUELICENSE-50: The XML element name MUST be explicitly defined!
29  // Otherwise, it would get derived from the class name, but this would break
30  // if the class name gets obfuscated, e.g. when using the ProGuard
31  // configuration from the TrueLicense Maven Archetype.
32  @XmlRootElement(name = "repository")
33  // #TRUELICENSE-52: Dito for the XML type. This enables objects of this class
34  // to participate in larger object graphs which the application wants to
35  // encode/decode to/from XML.
36  @XmlType(name = "repository")
37  public final class BasicRepository
38  extends RepositoryModel implements Repository {
39  
40      private final Base64 base64 = new Base64();
41  
42      @Override public final RepositoryModel model() { return this; }
43  
44      private @Nullable byte[] data(
45              final Codec codec,
46              final @CheckForNull String body) {
47          if (null == body) return null;
48          final @CheckForNull Charset cs = charset(codec);
49          return null != cs ? body.getBytes(cs) : decode(body);
50      }
51  
52      private @Nullable byte[] decode(@CheckForNull String body) {
53          return base64.decode(body);
54      }
55  
56      private @Nullable String body(
57              final Codec codec,
58              final @CheckForNull byte[] artifact) {
59          if (null == artifact) return null;
60          final @CheckForNull Charset cs = charset(codec);
61          return null != cs ? new String(artifact, cs) : encode(artifact);
62      }
63  
64      private @Nullable String encode(@CheckForNull byte[] data) {
65          return base64.encodeToString(data);
66      }
67  
68      @Override public final Artifactory sign(
69              final Codec codec,
70              final Signature engine,
71              final PrivateKey key,
72              final @Nullable Object artifact)
73      throws Exception {
74          final MemoryStore store = new MemoryStore();
75          codec.encode(store, artifact);
76          final byte[] artifactData = store.data();
77          engine.initSign(key);
78          engine.update(artifactData);
79          final byte[] signatureData = engine.sign();
80  
81          final String encodedArtifact = body(codec, artifactData);
82          final String encodedSignature = encode(signatureData);
83          final String signatureAlgorithm = engine.getAlgorithm();
84          final Artifactory artifactory = new EncodedArtifact(codec, store);
85  
86          final RepositoryModel model = model();
87          model.setArtifact(encodedArtifact);
88          model.setSignature(encodedSignature);
89          model.setAlgorithm(signatureAlgorithm);
90  
91          return artifactory;
92      }
93  
94      @Override public final Artifactory verify(
95              final Codec codec,
96              final Signature engine,
97              final PublicKey key)
98      throws Exception {
99          final RepositoryModel model = model();
100         if (!engine.getAlgorithm().equalsIgnoreCase(model.getAlgorithm()))
101             throw new IllegalArgumentException();
102         engine.initVerify(key);
103         final byte[] artifactData = data(codec, model.getArtifact());
104         engine.update(artifactData);
105         if (!engine.verify(decode(model.getSignature())))
106             throw new RepositoryIntegrityException();
107         return new EncodedArtifact(codec, artifactData);
108     }
109 }
