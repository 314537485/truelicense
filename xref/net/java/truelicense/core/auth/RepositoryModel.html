6   package net.java.truelicense.core.auth;
7   
8   import net.java.truelicense.core.util.Objects;
9   import net.java.truelicense.core.util.Strings;
10  
11  import javax.annotation.Nullable;
12  import javax.annotation.ParametersAreNullableByDefault;
13  import javax.xml.bind.annotation.XmlElement;
14  
15  import static java.util.Locale.ROOT;
16  
17  /**
18   * A model for storing authenticated objects (artifacts) as encoded strings.
19   * The content type and the content transfer encoding of the encoded artifact
20   * and encoded signature are defined externally.
21   * <p>
22   * This class is used by {@linkplain Repository repositories} to store
23   * artifacts for {@linkplain Repository#sign signing} and
24   * {@linkplain Repository#verify verifying}.
25   *
26   * @author Christian Schlichtherle
27   */
28  @ParametersAreNullableByDefault
29  @Nullable
30  public class RepositoryModel {
31  
32      private String artifact, signature, algorithm;
33  
34      /** Returns the encoded artifact. */
35      @XmlElement(required = true)
36      public final String getArtifact() { return artifact; }
37  
38      /** Sets the encoded artifact. */
39      public final void setArtifact(final String encodedArtifact) {
40          this.artifact = encodedArtifact;
41      }
42  
43      /** Returns the encoded signature. */
44      @XmlElement(required = true)
45      public final String getSignature() { return signature; }
46  
47      /** Sets the encoded signature. */
48      public void setSignature(final String encodedSignature) {
49          this.signature = encodedSignature;
50      }
51  
52      /** Returns the signature algorithm. */
53      @XmlElement(required = true)
54      public final String getAlgorithm() { return algorithm; }
55  
56      /** Sets the signature algorithm. */
57      public final void setAlgorithm(final String signatureAlgorithm) {
58          this.algorithm = signatureAlgorithm;
59      }
60  
61      /**
62       * Returns {@code true} if and only if the given object is an instance of
63       * {@code RepositoryModel} and it's encoded artifact, encoded signature and
64       * signature algorithm compare equal, whereby the case of the signature
65       * algorithm is ignored.
66       */
67      @Override
68      @SuppressWarnings("AccessingNonPublicFieldOfAnotherObject")
69      public boolean equals(final Object obj) {
70          if (this == obj) return true;
71          if (!(obj instanceof RepositoryModel)) return false;
72          final RepositoryModel that = (RepositoryModel) obj;
73          return  Objects.equals(this.getArtifact(), that.getArtifact()) &&
74                  Objects.equals(this.getSignature(), that.getSignature()) &&
75                  Strings.equalsIgnoreCase(this.getAlgorithm(), that.getAlgorithm());
76      }
77  
78      /** Returns a hash code which is consistent with {@link #equals}. */
79      @Override
80      public int hashCode() {
81          int c = 17;
82          c = 31 * c + Objects.hashCode(getArtifact());
83          c = 31 * c + Objects.hashCode(getSignature());
84          c = 31 * c + Objects.hashCode(Strings.toLowerCase(getAlgorithm(), ROOT));
85          return c;
86      }
87  }
