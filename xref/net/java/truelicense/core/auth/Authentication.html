6   package net.java.truelicense.core.auth;
7   
8   import javax.annotation.Nullable;
9   import net.java.truelicense.core.codec.Codec;
10  
11  /**
12   * Defines authentication services.
13   *
14   * @author Christian Schlichtherle
15   */
16  public interface Authentication extends AuthenticationParametersProvider {
17  
18      /**
19       * Encodes and signs the given {@code artifact} in the provided
20       * repository and returns an artifactory for decoding it.
21       * After calling this method, the repository typically gets encoded for
22       * future decoding and verification.
23       *
24       * @param  codec the codec for encoding the artifact.
25       *         It is an implementation detail whether or not encoding
26       *         {@code null} is supported or not.
27       * @param  repository the repository for encoding the artifact to.
28       * @param  artifact the nullable artifact to sign.
29       *         This may be {@code null} if and only if the {@code codec}
30       *         supports encoding it.
31       * @return An Artifactory for decoding the signed artifact in the
32       *         repository.
33       */
34      Artifactory sign(Codec codec, Repository repository, @Nullable Object artifact) throws Exception;
35  
36      /**
37       * Verifies the signature of the encoded artifact in the provided
38       * repository and returns an artifactory for decoding it.
39       * Calling this method generally requires prior initialization of the
40       * repository.
41       * Though initialization is a side effect of calling {@link #sign}, this is
42       * typically done by decoding the repository itself.
43       *
44       * @param  codec the codec for decoding the artifact.
45       *         It is an implementation detail whether or not decoding
46       *         {@code null} is supported or not.
47       * @param  repository the repository for decoding the artifact from.
48       * @return An Artifactory for decoding the verified artifact in the
49       *         repository.
50       * @throws RepositoryIntegrityException if the integrity of the repository
51       *         with its encoded artifact has been compromised.
52       */
53      Artifactory verify(Codec codec, Repository repository) throws Exception;
54  }
