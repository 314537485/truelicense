6   package net.java.truelicense.core.auth;
7   
8   import net.java.truelicense.core.util.Message;
9   
10  import java.security.GeneralSecurityException;
11  import java.util.Locale;
12  
13  import static net.java.truelicense.core.util.Objects.requireNonNull;
14  
15  /**
16   * Indicates that a notary could not access the private or public key in a
17   * key store due to insufficient or incorrect configuration parameters.
18   *
19   * @author Christian Schlichtherle
20   */
21  public class NotaryException extends GeneralSecurityException {
22  
23      private static final long serialVersionUID = 1L;
24  
25      private final Message msg;
26  
27      /**
28       * Constructs a notary exception with the given message.
29       *
30       * @param msg the message.
31       */
32      public NotaryException(final Message msg) {
33          this.msg = requireNonNull(msg);
34      }
35  
36      @Override
37      public String getMessage() { return msg.toString(Locale.ROOT); }
38  
39      @Override
40      public String getLocalizedMessage() { return msg.toString(); }
41  }
